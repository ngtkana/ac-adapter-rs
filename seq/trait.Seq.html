fn adjacent(self) -> Adjacent<Self, Self::Item> ⓘwhere
    Self::Item: Clone,

隣りあう 2 つの項のタプルを走査です。

Examples
use seq::Seq;
let mut iter = [0, 1, 2].iter().copied().adjacent();
assert_eq!(iter.next(), Some((0, 1)));
assert_eq!(iter.next(), Some((1, 2)));
assert_eq!(iter.next(), None);

要素が 1 個以下のときには空イテレータになります。

use seq::Seq;
let mut iter = [0].iter().adjacent();
assert_eq!(iter.next(), None);

fn grid_next(self, ij: (usize, usize), h: usize, w: usize) -> GridNext<Self> ⓘwhere
    Self: Iterator<Item = (i64, i64)>,

隣接するグリッドを走査できます。

Examples
use seq::Seq;
const KNIGHT: [(i64, i64); 8] = [
    (1, 2),
    (2, 1),
    (1, -2),
    (2, -1),
    (-1, 2),
    (-2, 1),
    (-1, -2),
    (-2, -1),
];
let mut result = KNIGHT
    .iter()
    .copied()
    .grid_next((1, 2), 4, 3)
    .collect::<Vec<_>>();
let expected = vec![(2, 0),(3, 1), (0, 0)];
assert_eq!(result, expected);
