fn prefix_sum(&mut self)

Source

fn prefix_sum_inv(&mut self)

Source

fn suffix_sum(&mut self)
where for<'a> T: AddAssign<&'a T>,

Source

fn suffix_sum_inv(&mut self)
where for<'a> T: SubAssign<&'a T>,

Replace $A_i$ with $A_i - A_{i+1}$.
