1/// Chunk by predicate.
2pub trait SliceChunks {
3    type Item;
4    /// Groups adjacent elements by a predicate.
5    /// (Rust 1.77.0)
6    fn chunk_by<F>(&self, f: F) -> SliceChunkBy<'_, Self::Item, F>
7    where
8        F: FnMut(&Self::Item, &Self::Item) -> bool;
9}
10impl<T> SliceChunks for [T] {
11    type Item = T;
12
13    fn chunk_by<F>(&self, f: F) -> SliceChunkBy<'_, Self::Item, F>
14    where
15        F: FnMut(&Self::Item, &Self::Item) -> bool,
16    {
17        SliceChunkBy { a: self, f }
18    }
19}
20
21pub struct SliceChunkBy<'a, T, F> {
22    a: &'a [T],
23    f: F,
24}
25impl<'a, T, F> Iterator for SliceChunkBy<'a, T, F>
26where
27    F: FnMut(&T, &T) -> bool,
28{
29    type Item = &'a [T];
30
31    fn next(&mut self) -> Option<Self::Item> {
32        let Self { a, f } = self;
33        if a.is_empty() {
34            return None;
35        }
36        let mut end = 1;
37        while end < a.len() && f(&a[end - 1], &a[end]) {
38            end += 1;
39        }
40        let (prefix, rest) = a.split_at(end);
41        self.a = rest;
42        Some(prefix)
43    }
44}
