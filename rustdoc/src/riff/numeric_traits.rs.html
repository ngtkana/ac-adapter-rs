23/// Unsigned integer types.
24pub trait Unsigned:
25    Sized
26    + PartialEq
27    + PartialOrd
28    + Debug
29    + Clone
30    + Copy
31    + Add<Output = Self>
32    + AddAssign
33    + Sub<Output = Self>
34    + SubAssign
35    + Mul<Output = Self>
36    + MulAssign
37    + Div<Output = Self>
38    + DivAssign
39    + BitAnd<Output = Self>
40    + BitAndAssign
41    + BitOr<Output = Self>
42    + BitOrAssign
43    + BitXor<Output = Self>
44    + BitXorAssign
45    + Shl<u32, Output = Self>
46    + ShlAssign<u32>
47    + Shr<u32, Output = Self>
48    + ShrAssign<u32>
49    + Not<Output = Self>
50{
51    const BITS: u32;
52    const MAX: Self;
53    const ZERO: Self;
54    const ONE: Self;
55    fn wrapping_neg(self) -> Self;
56    fn bit_length() -> u32 {
57        size_of::<Self>() as u32 * 8
58    }
59}

61macro_rules! impl_unsigned {
62    ($($T:ty),* $(,)?) => {$(
63        impl Unsigned for $T {
64            const BITS: u32 = <$T>::BITS;
65            const MAX: Self = <$T>::MAX;
66            const ZERO: Self = 0;
67            const ONE: Self = 1;
68            fn wrapping_neg(self) -> Self { self.wrapping_neg() }
69        }
70    )*}
71}
