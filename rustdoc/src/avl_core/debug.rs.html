3pub fn display<C: NodeMarker>(tree: &CoreTree<C>) -> String
4where
5    C::Data: std::fmt::Debug,
6{
7    fn display_recur<C: NodeMarker>(x: &Node<C>, s: &mut String, d: u8)
8    where
9        C::Data: std::fmt::Debug,
10    {
11        use std::fmt::Write;
12        if let Some(l) = x.left.as_ref() {
13            display_recur(l, s, d + 1);
14        }
15        writeln!(
16            s,
17            "{space1}●{space2} len={len} ht={ht}{rev} {data:?}",
18            space1 = " ".repeat(d as usize),
19            space2 = " ".repeat(4_usize.saturating_sub(d as usize)),
20            len = x.len,
21            ht = x.ht,
22            rev = if x.rev { "[rev] " } else { "" },
23            data = x.data,
24        )
25        .unwrap();
26        if let Some(r) = x.right.as_ref() {
27            display_recur(r, s, d + 1);
28        }
29    }
30    let Some(x) = tree.root.as_ref() else { return "(empty)".to_owned() };
31    let mut s = String::new();
32    display_recur(x, &mut s, 1);
33    s
34}

36pub fn validate<C: NodeMarker>(tree: &CoreTree<C>)
37where
38    C::Data: std::fmt::Debug,
39{
40    fn validate_recur<C: NodeMarker>(x: &Node<C>)
41    where
42        C::Data: std::fmt::Debug,
43    {
44        let lh = x.left.as_ref().map_or(0, |l| l.ht);
45        let rh = x.right.as_ref().map_or(0, |r| r.ht);
46        assert!(matches!(lh as i8 - rh as i8, -1..=1));
47        assert_eq!(lh.max(rh) + 1, x.ht);
48        if let Some(l) = x.left.as_ref() {
49            validate_recur(l);
50        }
51        if let Some(r) = x.right.as_ref() {
52            validate_recur(r);
53        }
54    }
55    let Some(x) = tree.root.as_deref() else { return };
56    validate_recur(x);
57}
