13/// 指数部分に使うためのトレイトです。すべての符号なし整数型に実装されています。
14pub trait Pow {
15    /// `*x != 0`
16    fn is_nonzero(&self) -> bool;
17    /// `*x != 1`
18    fn is_nonone(&self) -> bool;
19    /// `x & 1 == 1`
20    fn is_odd(&self) -> bool;
21    /// `self >>= 1`
22    fn shr1(&mut self);
23}
24
25/// aⁿ(x) を計算します。
26///
27/// # Requirements
28///
29/// - `T` が積と `U` への作用を持つ
30/// - `square` が `T` における２乗
31/// - `apply` が `T` の `U` への作用
32///
33///
34/// # Examples
35///
36/// ```
37/// use binary::operator_binary;
38///
39/// let a = 2;
40/// let n = 5_u32; // `i32` はコンパイルエラー
41/// let x = 42;
42/// let result = operator_binary(a, n, x, |&i| i * i, |&i, j| i * j);
43/// assert_eq!(result, 32 * x);
44/// ```
45pub fn operator_binary<T, U>(
46    mut a: T,
47    mut n: impl Pow,
48    mut x: U,
49    mut square: impl FnMut(&T) -> T,
50    mut apply: impl FnMut(&T, U) -> U,
51) -> U {
52    if n.is_nonzero() {
53        while n.is_nonone() {
54            if n.is_odd() {
55                x = apply(&a, x);
56            }
57            a = square(&a);
58            n.shr1();
59        }
60        x = apply(&a, x);
61    }
62    x
63}
64
65/// aⁿを計算します。
66///
67/// # Requirements
68///
69/// - `T` が積と単位元を持つ
70/// - `identity` が `T` の単位元
71/// - `mul` が `T` の積
72///
73///
74/// # Examples
75///
76/// ```
77/// use binary::value_binary;
78///
79/// let a = 3;
80/// let n = 5_u32; // `i32` はコンパイルエラー
81/// let result = value_binary(a, n, 1, |&i, &j| i * j);
82/// assert_eq!(result, 243);
83/// ```
84pub fn value_binary<T>(
85    mut a: T,
86    mut n: impl Pow,
87    identity: T,
88    mut mul: impl FnMut(&T, &T) -> T,
89) -> T {
90    let mut ans = identity;
91    if n.is_nonzero() {
92        while n.is_nonone() {
93            if n.is_odd() {
94                ans = mul(&a, &ans);
95            }
96            a = mul(&a, &a);
97            n.shr1();
98        }
99        ans = mul(&a, &ans);
100    }
101    ans
102}

104macro_rules! impl_pow {
105    ($($T:ty),* $(,)?) => {$(
106        impl Pow for $T {
107            fn is_nonzero(&self) -> bool {
108                *self != 0
109            }
110            fn is_nonone(&self) -> bool {
111                *self != 1
112            }
113            fn is_odd(&self) -> bool {
114                self & 1 == 1
115            }
116            fn shr1(&mut self) {
117                *self >>= 1;
118            }
119        }
120    )*}
121}

123impl_pow! {
124    u8, u16, u32, u64, u128, usize,
125}
