1//! # Manacher's algorithm
2
3/// Returns the palindrome array $A$ of the given string $s$.
4///
5/// # Definition
6///
7/// $A$ is an array of length $2n + 1$ where $n$ is the length of $s$.
8///
9/// $A_i = \max \left\\{
10///     r - l \middle |
11///     \begin{array}{l}
12///         0 \le l \le r \le n, \ l + r = i, \\\\
13///         s[l..r] \text{ is a palindrome}
14///     \end{array}
15/// \right\\}$.
16///
17/// # Example
18///
19/// If $s = \text{"mississippi"}$, then
20/// $A = [0, 1, 0, 1, 0, 1, 4, 1, 0, 7, 0, 1, 4, 1, 0, 1, 0, 1, 4, 1, 0, 1, 0]$.
21///
22/// ```
23/// # use manacher::manacher;
24/// let s = "mississippi";
25/// let a = manacher(s.as_bytes());
26/// assert_eq!(a, vec![
27///     0, 1, 0, 1, 0, 1, 4, 1, 0, 7, 0, 1, 4, 1, 0, 1, 0, 1, 4, 1, 0, 1, 0
28/// ]);
29/// ```
30///
31/// | $i$ | $s_{(i - 1) / 2}$ | $A_i$ | $s_{(i - A_i) / 2..(i + A_i) / 2}$ |
32/// | --- | ------ | ------ | --------- |
33/// | 0   |        | 0      |           |
34/// | 1   | m      | 1      | m         |
35/// | 2   |        | 0      |           |
36/// | 3   | i      | 1      | i         |
37/// | 4   |        | 0      |           |
38/// | 5   | s      | 1      | s         |
39/// | 6   |        | 4      | issi      |
40/// | 7   | s      | 1      | s         |
41/// | 8   |        | 0      |           |
42/// | 9   | i      | 7      | ississi   |
43/// | 10  |        | 0      |           |
44/// | 11  | s      | 1      | s         |
45/// | 12  |        | 4      | issi      |
46/// | 13  | s      | 1      | s         |
47/// | 14  |        | 0      |           |
48/// | 15  | i      | 1      | i         |
49/// | 16  |        | 0      |           |
50/// | 17  | p      | 1      | p         |
51/// | 18  |        | 4      | ippi      |
52/// | 19  | p      | 1      | p         |
53/// | 20  |        | 0      |           |
54/// | 21  | i      | 1      | i         |
55/// | 22  |        | 0      |           |
56pub fn manacher<T: Eq>(s: &[T]) -> Vec<usize> {
57    let n = s.len();
58    let mut a = vec![0; 2 * n + 1];
59    let mut i = 1;
60    let mut j = 1;
61    while i <= 2 * n {
62        while j < i && i + j < 2 * n && s[(i - j) / 2 - 1] == s[usize::midpoint(i, j)] {
63            j += 2;
64        }
65        a[i] = j;
66        if j == 0 {
67            i += 1;
68            j = 1;
69            continue;
70        }
71        let mut k = 1;
72        while k <= i && k + a[i - k] < j {
73            a[i + k] = a[i - k];
74            k += 1;
75        }
76        i += k;
77        j -= k;
78    }
79    a
80}

82#[cfg(test)]
83mod tests {
84    use super::*;
85    use rand::rngs::StdRng;
86    use rand::Rng;
87    use rand::SeedableRng;
88
89    #[test]
90    fn test_manacher() {
91        let mut rng = StdRng::seed_from_u64(42);
92        for _ in 0..200 {
93            let n = rng.gen_range(0..10);
94            let s = (0..n).map(|_| rng.gen_range(0..4)).collect::<Vec<_>>();
95            let result = manacher(&s);
96            for (i, &result) in result.iter().enumerate() {
97                let mut l = i / 2;
98                let mut r = i - l;
99                while 0 < l && r < n && s[l - 1] == s[r] {
100                    l -= 1;
101                    r += 1;
102                }
103                assert_eq!(result, r - l);
104            }
105        }
106    }
