1/// Return the multiplicative inverse of `x` modulo `P`.
2/// # Requirements
3/// - $P$ is odd and prime ($P ≥ 2³¹$)
4/// - $x < P$
5/// # Returns
6/// $x⁻¹ \mod P$
7/// # Postconditions
8/// - $0 ≤ \text{result} < P$
9pub(crate) fn mod_inv<const P: u64>(x: u64) -> u64 {
10    debug_assert!(P % 2 == 1);
11    debug_assert!(P < 1 << 31);
12    debug_assert!(x < P);
13    mod_inv_signed(x as i64, P as i64) as u64
14}
15/// Returns $a⁻¹ \mod m$.
16/// # Requirements
17/// - $m > 0, 0 < a < m$
18fn mod_inv_signed(a: i64, m: i64) -> i64 {
19    debug_assert!(a > 0);
20    debug_assert!(m > 0);
21    if a == 1 {
22        return 1;
23    }
24    m + (1 - m * mod_inv_signed(m % a, a)) / a
25}

27#[cfg(test)]
28mod tests {
29    use super::*;
30    use rand::rngs::StdRng;
31    use rand::Rng;
32    use rand::SeedableRng;
33
34    #[test]
35    fn test_mod_inverse() {
36        let mut rng = StdRng::seed_from_u64(42);
37        for _ in 0..256 {
38            let m = rng.gen_range(2..256);
39            let a = rng.gen_range(1..m);
40            if num::integer::gcd(a, m) != 1 {
41                continue;
42            }
43            let c = mod_inv_signed(a, m);
44            assert_eq!(a * c % m, 1, "a = {a}, c = {c}, ");
45            assert!((0..m).contains(&c));
46        }
47    }
