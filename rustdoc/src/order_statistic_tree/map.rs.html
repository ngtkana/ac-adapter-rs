10/// A trait for segment-tree style augmentation.
11///
12/// Defines a monoid structure over aggregated values computed from key-value pairs.
13pub trait Op {
14    /// The key type associated with this operation.
15    type Key;
16    /// The value type associated with this operation.
17    type Value;
18    /// The type of the aggregated/segment value.
19    type SegValue: Clone;
20
21    /// Returns the monoid identity element.
22    fn identity() -> Self::SegValue;
23
24    /// Lifts a key-value pair to a segment value.
25    fn to_seg_value(key: &Self::Key, value: &Self::Value) -> Self::SegValue;
26
27    /// Monoid multiplication (associative).
28    fn mul(lhs: &Self::SegValue, rhs: &Self::SegValue) -> Self::SegValue;
29}
30
31/// A no-op augmentation trait for non-augmented maps.
32/// SegValue is (), providing zero-cost abstraction.
33pub struct NoOp<K, V>(PhantomData<(K, V)>);
34
35impl<K, V> Op for NoOp<K, V> {
36    type Key = K;
37    type Value = V;
38    type SegValue = ();
39
40    fn identity() -> Self::SegValue {}
41    fn to_seg_value(_: &K, _: &V) -> Self::SegValue {}
42    fn mul((): &(), (): &()) -> Self::SegValue {}
43}

54/// An order-statistic map backed by a splay tree.
55///
56/// This data structure maintains key-value pairs in sorted order and supports
57/// order-statistic operations like `nth` and `binary_search`, all in amortized O(log n).
58///
59/// # Examples
60///
61/// ```
62/// use order_statistic_tree::OrderStatisticMap;
63///
64/// let mut map: OrderStatisticMap<i32, &str> = OrderStatisticMap::new();
65/// map.insert(3, "c");
66/// map.insert(1, "a");
67/// map.insert(2, "b");
68///
69/// assert_eq!(map.nth(0), (&1, &"a"));
70/// assert_eq!(map.get(&2), Some(&"b"));
71/// ```
72pub struct OrderStatisticMap<K, V, O: Op<Key = K, Value = V> = NoOp<K, V>> {
73    root: Cell<Option<NonNull<Node<K, V, O>>>>,
74}

143    fn new_node(
144        key: K,
145        value: V,
146        parent: Option<NonNull<Node<K, V, O>>>,
147    ) -> NonNull<Node<K, V, O>> {
148        let prod = O::to_seg_value(&key, &value);
149        let node = Box::into_raw(Box::new(Node {
150            key,
151            value,
152            parent,
153            left: None,
154            right: None,
155            len: 1,
156            prod,
157        }));
158        NonNull::new(node).unwrap()
159    }

161    /// Inserts a key-value pair into the map.
162    ///
163    /// If the map did not have this key present, `None` is returned.
164    ///
165    /// If the map did have this key present, the value is updated, and the old
166    /// value is returned. The key is not updated, though.
167    ///
168    /// # Examples
169    ///
170    /// ```
171    /// use order_statistic_tree::OrderStatisticMap;
172    ///
173    /// let mut map: OrderStatisticMap<i32, &str> = OrderStatisticMap::new();
174    /// assert_eq!(map.insert(1, "a"), None);
175    /// assert_eq!(map.insert(1, "b"), Some("a"));
176    /// ```
177    pub fn insert(&mut self, key: K, value: V) -> Option<V> {
178        match self.root.get() {
179            None => {
180                self.root.set(Some(Self::new_node(key, value, None)));
181                None
182            }
183            Some(root) => unsafe {
184                let mut current = root;
185                loop {
186                    let key_cmp = key.cmp(&(*current.as_ptr()).key);
187                    match key_cmp {
188                        std::cmp::Ordering::Less => {
189                            if let Some(left) = (*current.as_ptr()).left {
190                                current = left;
191                            } else {
192                                let new_node = Self::new_node(key, value, Some(current));
193                                (*current.as_ptr()).left = Some(new_node);
194                                current = new_node;
195                                break;
196                            }
197                        }
198                        std::cmp::Ordering::Greater => {
199                            if let Some(right) = (*current.as_ptr()).right {
200                                current = right;
201                            } else {
202                                let new_node = Self::new_node(key, value, Some(current));
203                                (*current.as_ptr()).right = Some(new_node);
204                                current = new_node;
205                                break;
206                            }
207                        }
208                        std::cmp::Ordering::Equal => {
209                            let old_value =
210                                std::mem::replace(&mut (*current.as_ptr()).value, value);
211                            (*current.as_ptr()).prod = O::to_seg_value(
212                                &(*current.as_ptr()).key,
213                                &(*current.as_ptr()).value,
214                            );
215                            current = splay(current);
216                            self.root.set(Some(current));
217                            return Some(old_value);
218                        }
219                    }
220                }
221                current = splay(current);
222                self.root.set(Some(current));
223                None
224            },
225        }
226    }

285    /// Returns an iterator over the map in sorted order.
286    ///
287    /// # Examples
288    ///
289    /// ```
290    /// use order_statistic_tree::OrderStatisticMap;
291    ///
292    /// let mut map: OrderStatisticMap<i32, &str> = OrderStatisticMap::new();
293    /// map.insert(3, "c");
294    /// map.insert(1, "a");
295    /// map.insert(2, "b");
296    ///
297    /// let entries: Vec<_> = map.iter().collect();
298    /// assert_eq!(entries, vec![(&1, &"a"), (&2, &"b"), (&3, &"c")]);
299    /// ```
300    pub fn iter(&self) -> Iter<'_, K, V, O> {
301        let mut stack = Vec::new();
302        if let Some(root) = self.root.get() {
303            let mut current = Some(root);
304            while let Some(c) = current {
305                stack.push(c);
306                current = unsafe { (*c.as_ptr()).left };
307            }
308        }
309        Iter {
310            stack,
311            _phantom: std::marker::PhantomData,
312        }
313    }

641    /// Concatenates `other` into `self`, requiring all keys in `self` to be strictly less than
642    /// all keys in `other`.
643    ///
644    /// After this operation, `other` becomes empty and `self` contains all key-value pairs
645    /// from both maps, sorted by key. This operation takes O(log n) amortized time.
646    ///
647    /// # Panics
648    ///
649    /// Panics if both maps are non-empty and the maximum key in `self` is not strictly less than
650    /// the minimum key in `other`.
651    ///
652    /// # Examples
653    ///
654    /// ```
655    /// use order_statistic_tree::OrderStatisticMap;
656    ///
657    /// let mut map1: OrderStatisticMap<i32, &str> = OrderStatisticMap::new();
658    /// map1.insert(1, "a");
659    /// map1.insert(2, "b");
660    ///
661    /// let mut map2: OrderStatisticMap<i32, &str> = OrderStatisticMap::new();
662    /// map2.insert(3, "c");
663    /// map2.insert(4, "d");
664    ///
665    /// map1.concat(&mut map2);
666    /// assert_eq!(map1.len(), 4);
667    /// assert!(map2.is_empty());
668    /// ```
669    pub fn concat(&mut self, other: &mut Self) {
670        let self_root = self.root.get();
671        let other_root = other.root.get();
672        match (self_root, other_root) {
673            (_, None) => {}
674            (None, Some(_)) => {
675                self.root.set(other_root);
676                other.root.set(None);
677            }
678            (Some(_), Some(_)) => {
679                let self_max = rightmost_and_splay(self).unwrap();
680                let other_min = leftmost_and_splay(other).unwrap();
681                unsafe {
682                    assert!(
683                        (*self_max.as_ptr()).key < (*other_min.as_ptr()).key,
684                        "OrderStatisticMap::concat requires all keys in self to be strictly less than all keys in other"
685                    );
686                }
687                let merged = node::merge_trees(Some(self_max), Some(other_min));
688                self.root.set(merged);
689                other.root.set(None);
690            }
691        }
692    }

694    /// Internal helper for splitting at a given index.
695    fn split_off_at(&mut self, idx: usize) -> Self {
696        let len = self.len();
697        assert!(idx <= len, "index out of bounds: idx={idx} len={len}");
698        if idx == len {
699            return Self::new();
700        }
701        if idx == 0 {
702            let new_map = Self {
703                root: Cell::new(self.root.get()),
704            };
705            self.root.set(None);
706            return new_map;
707        }
708        let node = nth_and_splay(self, idx);
709        unsafe {
710            let left = (*node.as_ptr()).left.take();
711            if let Some(l) = left {
712                (*l.as_ptr()).parent = None;
713            }
714            (*node.as_ptr()).parent = None;
715            (*node.as_ptr()).update();
716            self.root.set(left);
717        }
718        Self {
719            root: Cell::new(Some(node)),
720        }
721    }

723    /// Splits the map into two at the given split key, returning a new map containing all
724    /// key-value pairs with keys greater than or equal to the split key.
725    ///
726    /// This operation takes O(log n) amortized time. The original map retains all key-value
727    /// pairs with keys strictly less than the split key.

729    /// # Examples
730    ///
731    /// ```
732    /// use order_statistic_tree::OrderStatisticMap;

734    /// let mut map: OrderStatisticMap<i32, &str> = OrderStatisticMap::new();

794    /// Computes the aggregate value of the entire map using the configured Op.
795    ///
796    /// Returns the monoid product of all (key, value) pairs in the map.
797    /// For an empty map, returns `O::identity()`.
798    ///
799    /// This operation takes O(1) time by accessing the cached product at the root node.
800    ///
801    /// # Examples
802    ///
803    /// ```
804    /// # use order_statistic_tree::{OrderStatisticMap, Op};
805    /// # struct SumOp;
806    /// # impl Op for SumOp {
807    /// #     type Key = i32;
808    /// #     type Value = i32;
809    /// #     type SegValue = i64;
810    /// #     fn identity() -> i64 { 0 }
811    /// #     fn to_seg_value(key: &i32, value: &i32) -> i64 { (*key as i64) + (*value as i64) }
812    /// #     fn mul(lhs: &i64, rhs: &i64) -> i64 { lhs + rhs }
813    /// # }
814    /// let mut map: OrderStatisticMap<i32, i32, SumOp> = OrderStatisticMap::new();
815    /// map.insert(1, 10);
816    /// map.insert(2, 20);
817    /// assert_eq!(map.fold_all(), (1 + 10) + (2 + 20)); // 33
818    /// ```
819    pub fn fold_all(&self) -> O::SegValue {
820        self.root
821            .get()
822            .map_or_else(O::identity, |r| unsafe { (*r.as_ptr()).prod.clone() })
823    }

825    /// Computes the aggregate value over a key range.
826    ///
827    /// Returns the monoid product of all (key, value) pairs within the specified key range.
828    /// The range can be specified using any `RangeBounds<K>` (e.g., `1..3`, `1..=3`, `..5`).
829    /// For an empty range, returns `O::identity()`.
830    ///
831    /// This operation takes O(log n) amortized time via a top-down descent pattern.
832    ///
833    /// # Examples
834    ///
835    /// ```
836    /// # use order_statistic_tree::{OrderStatisticMap, Op};
837    /// # struct SumOp;
838    /// # impl Op for SumOp {
839    /// #     type Key = i32;
840    /// #     type Value = i32;
841    /// #     type SegValue = i64;
842    /// #     fn identity() -> i64 { 0 }
843    /// #     fn to_seg_value(key: &i32, value: &i32) -> i64 { (*key as i64) + (*value as i64) }
844    /// #     fn mul(lhs: &i64, rhs: &i64) -> i64 { lhs + rhs }
845    /// # }
846    /// let mut map: OrderStatisticMap<i32, i32, SumOp> = OrderStatisticMap::new();
847    /// map.insert(1, 10);
848    /// map.insert(2, 20);
849    /// map.insert(3, 30);
850    /// assert_eq!(map.fold_by_key(1..3), (1 + 10) + (2 + 20)); // 33
851    /// ```
852    pub fn fold_by_key<Q: Ord + ?Sized>(&self, range: impl RangeBounds<Q>) -> O::SegValue
853    where
854        K: Borrow<Q>,
855    {
856        let start = match range.start_bound() {
857            Bound::Included(k) => self.lower_bound(k),
858            Bound::Excluded(k) => self.upper_bound(k),
859            Bound::Unbounded => 0,
860        };
861        let end = match range.end_bound() {
862            Bound::Included(k) => self.upper_bound(k),
863            Bound::Excluded(k) => self.lower_bound(k),
864            Bound::Unbounded => self.len(),
865        };
866        if end <= start {
867            return O::identity();
868        }
869        self.fold_by_index(start..end)
870    }

872    /// Computes the aggregate value over an index range.
873    ///
874    /// Returns the monoid product of all (key, value) pairs at indices within the specified range.
875    /// Indices are in sorted order, starting from 0.
876    /// The range can be specified using any `RangeBounds<usize>` (e.g., `0..3`, `0..=2`, `..5`).
877    /// For an empty range, returns `O::identity()`.
878    ///
879    /// # Panics
880    ///
881    /// Panics if the range exceeds the bounds of the map.
882    ///
883    /// # Examples
884    ///
885    /// ```
886    /// # use order_statistic_tree::{OrderStatisticMap, Op};
887    /// # struct SumOp;
888    /// # impl Op for SumOp {
889    /// #     type Key = i32;
890    /// #     type Value = i32;
891    /// #     type SegValue = i64;
892    /// #     fn identity() -> i64 { 0 }
893    /// #     fn to_seg_value(key: &i32, value: &i32) -> i64 { (*key as i64) + (*value as i64) }
894    /// #     fn mul(lhs: &i64, rhs: &i64) -> i64 { lhs + rhs }
895    /// # }
896    /// let mut map: OrderStatisticMap<i32, i32, SumOp> = OrderStatisticMap::new();
897    /// map.insert(1, 10);
898    /// map.insert(2, 20);
899    /// map.insert(3, 30);
900    /// assert_eq!(map.fold_by_index(0..2), (1 + 10) + (2 + 20)); // 33
901    /// ```
902    pub fn fold_by_index(&self, range: impl RangeBounds<usize>) -> O::SegValue {
903        let len = self.len();
904        let (start, end) = to_half_open_range(range, len);
905        assert!(
906            start <= end && end <= len,
907            "range out of bounds: {start}..{end}, len={len}"
908        );
909
910        if start == end {
911            return O::identity();
912        }
913        if start == 0 && end == len {
914            return self.fold_all();
915        }
916
917        let (left, rest) = node::split_at_index(self.root.get(), start);
918        let (mid, right) = node::split_at_index(rest, end - start);
919        let result = mid.map_or_else(O::identity, |m| unsafe { (*m.as_ptr()).prod.clone() });
920        self.root
921            .set(node::merge_trees(node::merge_trees(left, mid), right));
922        result
923    }

925    /// Returns the maximum index `r` in `[l, len()]` such that `f(fold_by_index(l..r))` returns `true`.
926    ///
927    /// The function `f` must be monotonic: once it returns `false` for some `r`,
928    /// it must return `false` for all larger `r`. Additionally, `f(O::identity())` must be `true`.
929    ///
930    /// This operation takes O(log n) amortized time.
931    ///
932    /// # Panics
933    ///
934    /// Panics if `l > len()`.
935    ///
936    /// # Examples
937    ///
938    /// ```
939    /// # use order_statistic_tree::{OrderStatisticMap, Op};
940    /// # struct SumOp;
941    /// # impl Op for SumOp {
942    /// #     type Key = i32;
943    /// #     type Value = i32;
944    /// #     type SegValue = i64;
945    /// #     fn identity() -> i64 { 0 }
946    /// #     fn to_seg_value(key: &i32, value: &i32) -> i64 { (*key as i64) + (*value as i64) }
947    /// #     fn mul(lhs: &i64, rhs: &i64) -> i64 { lhs + rhs }
948    /// # }
949    /// let mut map: OrderStatisticMap<i32, i32, SumOp> = OrderStatisticMap::new();
950    /// map.insert(1, 10);
951    /// map.insert(2, 20);
952    /// map.insert(3, 30);
953    /// assert_eq!(map.max_right(0, |&s| s <= 30), 1);
954    /// assert_eq!(map.max_right(0, |&s| s <= 60), 2);
955    /// ```
956    pub fn max_right<F>(&self, l: usize, f: F) -> usize
957    where
958        F: Fn(&O::SegValue) -> bool,
959    {
960        let len = self.len();
961        assert!(l <= len, "index out of bounds: l={l} len={len}");
962        if l == len {
963            return len;
964        }
965        max_right_and_splay(self, l, f)
966    }

969    ///
970    /// The function `f` must be monotonic: once it returns `false` for some `l`,
971    /// it must return `false` for all smaller `l`. Additionally, `f(O::identity())` must be `true`.
972    ///
973    /// This operation takes O(log n) amortized time.
974    ///
975    /// # Panics
976    ///
977    /// Panics if `r > len()`.
978    ///
979    /// # Examples
980    ///
981    /// ```

990    /// #     fn mul(lhs: &i64, rhs: &i64) -> i64 { lhs + rhs }
991    /// # }

993    /// map.insert(1, 10);
994    /// map.insert(2, 20);
995    /// map.insert(3, 30);
996    /// assert_eq!(map.min_left(3, |&s| s <= 30), 3);
997    /// assert_eq!(map.min_left(3, |&s| s <= 60), 1);
998    /// ```
999    pub fn min_left<F>(&self, r: usize, f: F) -> usize

1001        F: Fn(&O::SegValue) -> bool,

1008        if r == 0 {
1009            return 0;
1010        }

1012    }

1064/// An iterator over the entries of an [`OrderStatisticMap`], yielding key-value pairs in sorted order.
1065///
1066/// This iterator is returned by the [`OrderStatisticMap::iter`] method.
1067///
1068/// # Examples
1069///
1070/// ```
1071/// use order_statistic_tree::OrderStatisticMap;
1072///
1073/// let mut map: OrderStatisticMap<i32, &str> = OrderStatisticMap::new();
1074/// map.insert(1, "a");
1075/// map.insert(2, "b");
1076///
1077/// for (key, value) in map.iter() {
1078///     println!("{}: {}", key, value);
1079/// }
1080/// ```
1081pub struct Iter<'a, K, V, O: Op<Key = K, Value = V>> {
1082    stack: Vec<NonNull<Node<K, V, O>>>,
1083    _phantom: std::marker::PhantomData<&'a Node<K, V, O>>,
1084}
1085
1086impl<'a, K, V, O: Op<Key = K, Value = V>> Iterator for Iter<'a, K, V, O> {
1087    type Item = (&'a K, &'a V);
1088
1089    fn next(&mut self) -> Option<Self::Item> {
1090        let node = self.stack.pop()?;
1091        unsafe {
1092            if let Some(right) = (*node.as_ptr()).right {
1093                let mut current = Some(right);
1094                while let Some(c) = current {
1095                    self.stack.push(c);
1096                    current = (*c.as_ptr()).left;
1097                }
1098            }
1099            Some((&(*node.as_ptr()).key, &(*node.as_ptr()).value))
1100        }
1101    }
1102}

1158fn to_half_open_range<R: RangeBounds<usize>>(range: R, len: usize) -> (usize, usize) {
1159    let start = match range.start_bound() {
1160        Bound::Included(&s) => s,
1161        Bound::Excluded(&s) => s + 1,
1162        Bound::Unbounded => 0,
1163    };
1164    let end = match range.end_bound() {
1165        Bound::Included(&e) => e + 1,
1166        Bound::Excluded(&e) => e,
1167        Bound::Unbounded => len,
1168    };
1169    (start, end)
