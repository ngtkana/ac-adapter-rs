3/// Returns a `Vec` of the divisors. The divisors is guaranteed to be sorted in ascending order.
4///
5/// # Example
6///
7/// Basic usage:
8/// ```
9/// use trial::divisors;
10///
11/// assert_eq!(divisors(12u32), vec![1, 2, 3, 4, 6, 12]);
12/// ```
13pub fn divisors<T: Value>(n: T) -> Vec<T> {
14    let mut former = Vec::new();
15    let mut latter = Vec::new();
16    let mut d = T::one();
17    while d * d <= n {
18        if d.divides(n) {
19            former.push(d);
20            if d * d != n {
21                latter.push(n / d);
22            }
23        }
24        d.increment();
25    }
26    former.extend(latter.into_iter().rev());
27    former
28}

30/// Takes an unsigned integer and returns an iterator to yield all the divisors_unordered of it.
31///
32/// # Note
33///
34/// The order of the return value is not necessarily ascending, but ordered in "alternating" order.
35/// (eg. 1, 36, 2, 18, 3, 12, 4, 9)
36///
37/// # Example
38///
39/// Basic usage:
40/// ```
41/// use trial::divisors_unordered;
42///
43/// let mut iter = divisors_unordered(12u32);
44/// assert_eq!(iter.next(), Some(1));
45/// assert_eq!(iter.next(), Some(12));
46/// assert_eq!(iter.next(), Some(2));
47/// assert_eq!(iter.next(), Some(6));
48/// assert_eq!(iter.next(), Some(3));
49/// assert_eq!(iter.next(), Some(4));
50/// assert_eq!(iter.next(), None);
51/// ```
52pub fn divisors_unordered<T: Value>(n: T) -> Divisors<T> {
53    Divisors {
54        n,
55        d: T::zero(),
56        rev: false,
57    }
58}
59/// [See the document of a function `divisors_unordered`](`divisors_unordered`)
60pub struct Divisors<T> {
61    n: T,
62    d: T,
63    rev: bool,
64}
65impl<T: Value> Iterator for Divisors<T> {
66    type Item = T;
67
68    fn next(&mut self) -> Option<Self::Item> {
69        let Self { n, d, rev } = self;
70        if *rev {
71            *rev = false;
72            Some(*n / *d)
73        } else {
74            (*d).increment();
75            while *d * *d <= *n && !(*d).divides(*n) {
76                (*d).increment();
77            }
78            if *n < *d * *d {
79                None
80            } else {
81                if *n != *d * *d {
82                    *rev = true;
83                }
84                Some(*d)
85            }
86        }
87    }
88}
