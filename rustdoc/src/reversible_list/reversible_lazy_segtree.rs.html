22struct Marker<O> {

25#[allow(dead_code)]
26struct Data<O: Op> {
27    value: O::Value,
28    sum: O::Value,
29    op: O::Operator,
30}
31impl<O: Op> Debug for Data<O> {
32    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
33        f.debug_struct("Data")
34            .field("value", &self.value)
35            .field("sum", &self.sum)

40impl<O: Op> NodeMarker for Marker<O> {
41    type Data = Data<O>;
42
43    fn update(node: &mut Node<Self>) {
44        let mut sum = node.data.value.clone();
45        if let Some(l) = node.left.as_deref() {
46            sum = O::mul(&l.data.sum, &sum);
47        }
48        if let Some(r) = node.right.as_deref() {
49            sum = O::mul(&sum, &r.data.sum);
50        }
51        node.data.sum = sum;
52    }
53
54    fn push(node: &mut Node<Self>) {
55        if node.data.op != O::nop() {
56            let op = std::mem::replace(&mut node.data.op, O::nop());
57            if let Some(l) = node.left.as_deref_mut() {
58                l.data.sum = O::op(&op, &l.data.sum);
59                l.data.value = O::op(&op, &l.data.value);
60                l.data.op = O::compose(&op, &l.data.op);
61            }
62            if let Some(r) = node.right.as_deref_mut() {
63                r.data.sum = O::op(&op, &r.data.sum);
64                r.data.value = O::op(&op, &r.data.value);
65                r.data.op = O::compose(&op, &r.data.op);
66            }
67        }
68    }
