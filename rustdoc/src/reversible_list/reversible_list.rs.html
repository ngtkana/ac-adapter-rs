83    #[derive(Debug)]
84    enum Query {
85        Insert { index: usize, value: i32 },
86        Remove { index: usize },
87        Reverse { start: usize, end: usize },
88    }
89
90    #[test]
91    fn test() {
92        let mut rng = StdRng::seed_from_u64(42);
93        for tid in 1..=200 {
94            let q = 50;
95            let value_lim = 10;
96            let len_max = rng.gen_range(5..=50);
97            let mut n = 0usize;
98            let mut rlist = ReversibleList::new();
99            let mut vec = vec![];
100            #[allow(clippy::explicit_counter_loop)]
101            for qid in 1..=q {
102                let query = match rng.gen_range(0..3) {
103                    0 => {
104                        let mut start = rng.gen_range(0..=n + 1);
105                        let mut end = rng.gen_range(0..=n);
106                        if start > end {
107                            (start, end) = (end, start - 1);
108                        }
109                        Query::Reverse { start, end }
110                    }
111                    1..=2 => {
112                        if rng.gen_ratio(n as u32, len_max) {
113                            let index = rng.gen_range(0..n);
114                            Query::Remove { index }
115                        } else {
116                            let index = rng.gen_range(0..=n);
117                            let value = rng.gen_range(0..value_lim);
118                            Query::Insert { index, value }
119                        }
120                    }
121                    _ => unreachable!(),
122                };
123                eprintln!("Query #{tid}.{qid}: {query:?}");
124                match query {
125                    Query::Insert { index, value } => {
126                        rlist.insert(index, value);
127                        vec.insert(index, value);
128                        n += 1;
129                    }
130                    Query::Remove { index } => {
131                        let result = rlist.remove(index);
132                        let expected = vec.remove(index);
133                        assert_eq!(result, expected);
134                        n -= 1;
135                    }
136                    Query::Reverse { start, end } => {
137                        rlist.reverse(start, end);
138                        vec[start..end].reverse();
139                    }
140                }
141                eprintln!("{}", display(rlist.root.as_deref()));
142                let result = collect(rlist.root.as_deref());
143                eprintln!("   vec: {:?}", &vec);
144                eprintln!(" rlist: {:?}", &result);
145                assert_eq!(&vec, &result);
146                validate(rlist.root.as_deref());
147                eprintln!();
148            }
149        }
150    }
