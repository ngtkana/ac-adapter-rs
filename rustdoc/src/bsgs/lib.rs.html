1//! Baby-step giant-step により、巡回群上の離散対数をします。
2//!
3//! # 使い方
4//!
5//! [`Bsgs::new`] を使いましょう。
6//!
7//! ```
8//! use bsgs::Bsgs;
9//!
10//! // ℤ / 10 ℤ の乗法群は、巡回群 { 1, 3, 9, 7 }
11//! let bsgs = Bsgs::new(3, 4, |x, y| x * y % 10, |x| x * x * x % 10, || 1);
12//!
13//! assert_eq!(bsgs.log(1), Some(0));
14//! assert_eq!(bsgs.log(3), Some(1));
15//! assert_eq!(bsgs.log(9), Some(2));
16//! assert_eq!(bsgs.log(7), Some(3));
17//!
18//! // 存在しないものを指定すると、None です。
19//! assert_eq!(bsgs.log(0), None);
20//! assert_eq!(bsgs.log(2), None);
21//!
22//! // 原子根でない場合も、位数さえちゃんとしていればうごきます。
23//! let bsgs = Bsgs::new(9, 2, |x, y| x * y % 10, |x| x * x * x % 10, || 1);
24//! assert_eq!(bsgs.log(1), Some(0));
25//! assert_eq!(bsgs.log(9), Some(1));
26//!
27//! assert_eq!(bsgs.log(0), None);
28//! assert_eq!(bsgs.log(2), None);
29//! assert_eq!(bsgs.log(3), None);
30//! assert_eq!(bsgs.log(7), None);
31//! ```
32//!
33//!
34//! # 仕様検討
35//!
36//! * ちなみに位数の上界は、探索の打ち切りに用いています。
37//! * 群の演算は、モジュラス等が動的に与えられる可能性を考えて、型ではなくオブジェクトにしました。
38//! * たいてい ℤ / n ℤ
39//!
40//! の乗法群にしか使わない気がするのですが、それようのユーティルがうまく作れず……
41use std::collections::HashMap;
42use std::fmt::Debug;
43use std::fmt::Formatter;
44use std::fmt::Result;
45use std::hash::Hash;
46use std::iter::successors;

48/// Baby-stpp giant-step のソルバーです。
49#[derive(Clone)]
50pub struct Bsgs<T, Mul> {
51    generator: T,
52    ord_upper_bound: u64,
53    sqrt: u64,
54    giant_step_inverse: T,
55    map: HashMap<T, u64>,
56    mul: Mul,
57}
58impl<T, Mul> Bsgs<T, Mul>
59where
60    T: Copy + Hash + Eq,
61    Mul: Fn(T, T) -> T,
62{
63    /// 新しい BSGS ソルバーを構築します。
64    ///
65    /// # Parameters
66    ///
67    /// * `generator`: 生成子
68    /// * `ord_upper_bound`: 生成子の位数の上界
69    /// * `mul`, `inv`, `id`: 群の演算
70    ///
71    /// # 要件
72    ///
73    /// `(generator, mul, inv, id)` が巡回群の演算とその生成子
74    ///
75    ///
76    /// # 計算量
77    ///
78    /// Θ ( √ ( `ord_upper_bound` ) ) 回の `mul` の呼び出しと、2 回の `id` の呼び出し
79    ///
80    ///
81    /// # 計算しているもの
82    ///
83    /// * `sqrt`: `ord_upper_bound` の平方根
84    /// * `map`: `generator` の `0..sqrt` 乗
85    /// * `giant_step_inverse`: `generator` の `-sqrt` 乗
86    pub fn new<Inv, Id>(generator: T, ord_upper_bound: u64, mul: Mul, inv: Inv, id: Id) -> Self
87    where
88        Id: Fn() -> T,
89        Inv: Fn(T) -> T,
90    {
91        let sqrt = sqrt(ord_upper_bound);
92        let map = successors(Some(id()), |&acc| Some(mul(acc, generator)))
93            .take(sqrt as usize)
94            .enumerate()
95            .map(|(i, x)| (x, i as u64))
96            .collect::<HashMap<_, _>>();
97        let giant_step_inverse = inv(binary(generator, sqrt, id(), &mul));
98        Self {
99            generator,
100            ord_upper_bound,
101            sqrt,
102            giant_step_inverse,
103            map,
104            mul,
105        }
106    }
107
108    /// `x` の離散対数が存在すれば返し、存在しなければ `None` を返します。
109    ///
110    /// # 計算量
111    ///
112    /// Θ ( √ ( `ord_upper_bound` ) ) 回の `mul` の呼び出し
113    pub fn log(&self, mut x: T) -> Option<u64> {
114        let mut ans = 0;
115        while ans < self.ord_upper_bound {
116            match self.map.get(&x) {
117                None => {
118                    ans += self.sqrt;
119                    x = (self.mul)(x, self.giant_step_inverse);
120                }
121                Some(&i) => return Some(ans + i),
122            }
123        }
124        None
125    }
126}

137    }

140fn binary<T: Copy>(mut a: T, mut b: u64, init: T, f: impl Fn(T, T) -> T) -> T {
141    let mut ans = init;
142    if b != 0 {
143        while b != 1 {
144            if (b & 1) == 1 {
145                ans = f(ans, a);
146            }
147            a = f(a, a);
148            b /= 2;
149        }
150        ans = f(ans, a);
151    }
152    ans
153}

155fn sqrt(a: u64) -> u64 {
156    let f = |x: u64| (a / x + x) >> 1;
157    let mut x = (a as f64).sqrt() as u64;
158    let mut xn = f(x);
159    while x < xn {
160        x = xn;
161        xn = f(x);
162    }
163    while xn < x {
164        x = xn;
165        xn = f(x);
166    }
167    x
168}

170#[cfg(test)]
171mod tests {
172    use super::binary;
173    use super::Bsgs;
174
175    #[test]
176    fn test_additive() {
177        for n in 2..40 {
178            let add = |x, y| (x + y) % n;
179            let minus = |x| (n - x) % n;
180            for g in 0..n {
181                let ord = n / gcd(g, n);
182                let bsgs = Bsgs::new(g, ord, add, minus, || 0);
183                let mut j = 0;
184                for i in 0..ord {
185                    let k = bsgs.log(j).unwrap();
186                    assert_eq!(i, k);
187                    j = add(j, g);
188                }
189            }
190        }
191    }
192
193    #[test]
194    fn test_multiplicative() {
195        for n in 2..40 {
196            let mul = |x, y| (x * y) % n;
197            let pow = |x, y| binary(x, y, 1, mul);
198            let ord = totient(n);
199            let inv = |x| pow(x, ord - 1);
200            for g in (2..n)
201                .filter(|&g| gcd(n, g) == 1 && (1..ord).all(|p| pow(g, p) != 1) && pow(g, ord) == 1)
202            {
203                let bsgs = Bsgs::new(g, ord, mul, inv, || 1);
204                let mut j = 1;
205                for i in 0..ord {
206                    let k = bsgs.log(j).unwrap();
207                    assert_eq!(i, k);
208                    j = mul(j, g);
209                }
210            }
211        }
212    }

214    fn gcd(x: u64, y: u64) -> u64 {
215        if x == 0 {
216            y
217        } else {
218            gcd(y % x, x)
219        }
220    }
221
222    fn totient(mut n: u64) -> u64 {
223        let mut ans = n;
224        for p in 2.. {
225            if n < p * p {
226                break;
227            }
228            if n % p == 0 {
229                ans -= ans / p;
230                while n % p == 0 {
231                    n /= p;
232                }
233            }
234        }
235        if n != 1 {
236            ans -= ans / n;
237        }
238        ans
239    }
