1//! Suffix Array を計算します。
2
3/// Surrix Array を計算します。
4///
5/// TODO: 実装を短くできるような気がします。
6///
7/// # Examples
8///
9/// ```
10/// use suffix_array::suffix_array;
11/// let s = "abracadabra";
12/// let sa = suffix_array(s.as_bytes());
13/// assert_eq!(sa, vec![10, 7, 0, 3, 5, 8, 1, 4, 6, 9, 2]);
14/// ```
15pub fn suffix_array<T: Ord>(s: &[T]) -> Vec<usize> {
16    let n = s.len();
17    let mut ord: Vec<usize> = (0..n).collect();
18    ord.sort_by_key(|&i| &s[i]);
19    let mut cmp = vec![0; n];
20    for i in 1..n {
21        cmp[i] = if s[ord[i - 1]] == s[ord[i]] { cmp[i - 1] } else { cmp[i - 1] + 1 };
22    }
23    for d in std::iter::successors(Some(1), |x| Some(x * 2)).take_while(|&x| x <= n) {
24        let mut ord_inverse = vec![0; n];
25        (0..n).for_each(|i| ord_inverse[ord[i]] = i);
26        let cmp_count = *cmp.last().unwrap() + 1;
27        let mut pos = vec![0; cmp_count];
28        for (i, &c) in cmp.iter().enumerate().rev() {
29            pos[c] = i;
30        }
31
32        let mut ord_swp = vec![0; n];
33        let mut insert = |i| {
34            let c = cmp[ord_inverse[i]];
35            ord_swp[pos[c]] = i;
36            pos[c] += 1;
37        };
38        (n - d..n).for_each(&mut insert);
39        ord.iter()
40            .filter(|&&i| d <= i)
41            .map(|i| i - d)
42            .for_each(insert);
43
44        let mut cmp_swp = vec![0; n];
45        for i in 1..n {
46            let l = ord_swp[i - 1];
47            let r = ord_swp[i];
48            cmp_swp[i] = if cmp[ord_inverse[l]] == cmp[ord_inverse[r]]
49                && ord_inverse.get(l + d).map_or(n, |&ld| cmp[ld])
50                    == ord_inverse.get(r + d).map_or(n, |&rd| cmp[rd])
51            {
52                cmp_swp[i - 1]
53            } else {
54                cmp_swp[i - 1] + 1
55            };
56        }
57
58        ord = ord_swp;
59        cmp = cmp_swp;
60    }
61    ord
62}

64/// LCP 配列を計算します。
65///
66/// # 戻り値
67///
68/// `s[sa[i]..]` と `s[sa[i + 1]..]` の最小共通部分列のの長さ `lcp[i]` を並べた、長さ `n - 1`
69/// の配列 `lcp` を返します。
70pub fn lcp_array<T: Ord>(s: &[T], sa: &[usize]) -> Vec<usize> {
71    assert_eq!(s.len(), sa.len());
72    assert!(!s.is_empty());
73    assert!(!sa.is_empty());
74
75    let n = s.len();
76    let rnk = make_rank(sa);
77    let mut h = 0_usize;
78    let mut lcp = vec![0; n - 1];
79    for (i, &r) in rnk.iter().enumerate() {
80        h = h.saturating_sub(1);
81        if r != 0 {
82            let j = sa[r - 1];
83            h = s[i..]
84                .iter()
85                .zip(&s[j..])
86                .position(|(c, d)| c != d)
87                .unwrap_or_else(|| s[i..].len().min(s[j..].len()));
88            lcp[r - 1] = h;
89        }
90    }
91    lcp
92}

94fn make_rank(a: &[usize]) -> Vec<usize> {
95    let n = a.len();
96    let mut b = vec![n; n];
97    for (i, &x) in a.iter().enumerate() {
98        assert_eq!(b[x], n);
99        b[x] = i;
100    }
101    b
102}

111    fn sa_brute<T: Ord>(s: &[T]) -> Vec<usize> {
112        let mut ord: Vec<_> = (0..s.len()).collect();
113        ord.sort_by_key(|&i| &s[i..]);
114        ord
115    }
116
117    fn lcp_brute<T: Ord>(s: &[T], sa: &[usize]) -> Vec<usize> {
118        let n = s.len();
119        (0..n - 1)
120            .map(|i| {
121                s[sa[i]..]
122                    .iter()
123                    .zip(s[sa[i + 1]..].iter())
124                    .take_while(|&(c, d)| c == d)
125                    .count()
126            })
127            .collect()
128    }
