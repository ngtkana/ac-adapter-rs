1//! Stable partition algorithm
2//!
3//! See [`partition`] for details.
4
5/// Partitions a mutable slice in-place, stably.
6///
7/// Rearranges elements such that those for which `pred` returns `true` are placed before
8/// those for which `pred` returns `false`, while preserving the relative order within each group.
9///
10/// # Time Complexity
11/// O(n), where n is the length of the slice.
12///
13/// # Extra Memory
14/// O(n) for an auxiliary buffer that temporarily holds elements.
15///
16/// # Stability
17/// The relative order of elements for which `pred` returns `true` is preserved.
18/// The relative order of elements for which `pred` returns `false` is also preserved.
19///
20/// # Returns
21/// The number of elements for which `pred` returned `true`.
22///
23/// # Examples
24///
25/// Basic usage:
26/// ```
27/// let mut v = vec![1, 2, 3, 4, 5, 6];
28/// let n = partition::partition(&mut v, |&x| x % 2 == 0);
29/// assert_eq!(n, 3);
30/// assert_eq!(v[0..n], [2, 4, 6]);
31/// assert_eq!(v[n..], [1, 3, 5]);
32/// ```
33///
34/// Stability with duplicate keys:
35/// ```
36/// let mut pairs = vec![(2, 'a'), (1, 'b'), (2, 'c'), (1, 'd')];
37/// let n = partition::partition(&mut pairs, |&(k, _)| k == 2);
38/// assert_eq!(n, 2);
39/// // Elements with key 2 come first, and their relative order (a before c) is preserved.
40/// assert_eq!(&pairs[0..n], &[(2, 'a'), (2, 'c')]);
41/// // Elements with key 1 come next, and their relative order (b before d) is preserved.
42/// assert_eq!(&pairs[n..], &[(1, 'b'), (1, 'd')]);
43/// ```
44pub fn partition<T>(slice: &mut [T], mut pred: impl FnMut(&T) -> bool) -> usize {
45    let n = slice.len();
46    let mut guard = PartitionGuard {
47        slice,
48        read_idx: 0,
49        write_idx: 0,
50        aux: Vec::with_capacity(n),
51    };
52
53    while guard.read_idx < n {
54        let i = guard.read_idx;
55        let keep = pred(&guard.slice[i]);
56        // SAFETY: i < n == guard.slice.len(). read_idx is strictly increasing,
57        // so each element at index i is read exactly once.
58        let value = unsafe { guard.slice.as_ptr().add(i).read() };
59        if keep {
60            // SAFETY: write_idx <= i < n. The destination at write_idx is either
61            // uninitialized (write_idx == i) or contains a stale duplicate
62            // (write_idx < i, previously moved to aux). write() does not drop the
63            // old value, so it is safe.
64            unsafe {
65                guard.slice.as_mut_ptr().add(guard.write_idx).write(value);
66            }
67            guard.write_idx += 1;
68        } else {
69            guard.aux.push(value);
70        }
71        guard.read_idx += 1;
72    }
73
74    let true_count = guard.write_idx;
75    for (offset, value) in guard.aux.drain(..).enumerate() {
76        // SAFETY: true_count + offset < n (loop invariant: aux.len() == read_idx - write_idx).
77        // The destination at true_count + offset is a stale duplicate that was read and moved to aux.
78        unsafe {
79            guard
80                .slice
81                .as_mut_ptr()
82                .add(true_count + offset)
83                .write(value);
84        }
85    }
86    guard.write_idx = n;
87
88    true_count
89}
90
91struct PartitionGuard<'a, T> {
92    slice: &'a mut [T],
93    read_idx: usize,
94    write_idx: usize,
95    aux: Vec<T>,
96}
97
98impl<T> Drop for PartitionGuard<'_, T> {
99    fn drop(&mut self) {
100        // On panic during pred(), slice[write_idx..read_idx) contains stale duplicates
101        // (the actual values are in aux). We restore the invariant by writing them back.
102        for idx in (self.write_idx..self.read_idx).rev() {
103            let value = self
104                .aux
105                .pop()
106                .expect("invariant: aux.len() == read_idx - write_idx");
107            // SAFETY: idx < read_idx <= slice.len(), and aux.pop() gives exactly the
108            // stale duplicate at this position (1-to-1 correspondence via loop invariant).
109            unsafe {
110                self.slice.as_mut_ptr().add(idx).write(value);
111            }
112        }
113    }
114}

172    fn stable_partition_brute<T: Clone>(v: &[T], pred: impl Fn(&T) -> bool) -> Vec<T> {
173        let mut true_part = Vec::new();
174        let mut false_part = Vec::new();
175        for item in v {
176            if pred(item) {
177                true_part.push(item.clone());
178            } else {
179                false_part.push(item.clone());
180            }
181        }
182        true_part.extend(false_part);
183        true_part
184    }

186    #[test]
187    fn test_random() {
188        use rand::prelude::StdRng;
189        use rand::Rng;
190        use rand::SeedableRng;
191
192        let mut rng = StdRng::seed_from_u64(42);
193        for _ in 0..150 {
194            let len = rng.gen_range(0..50);
195            let values: Vec<u32> = (0..len).map(|_| rng.gen()).collect();
196            let threshold: u32 = rng.gen();
197            let pred = |&x: &u32| x < threshold;
198
199            // Tag each value with its original index
200            let tagged: Vec<(u32, usize)> = values
201                .iter()
202                .enumerate()
203                .map(|(idx, &val)| (val, idx))
204                .collect();
205
206            let mut partition_result = tagged.clone();
207            let n = partition(&mut partition_result, |&(v, _)| pred(&v));
208
209            // Brute-force reference
210            let brute_result = stable_partition_brute(&tagged, |&(v, _)| pred(&v));
211
212            // Check the result matches the brute-force
213            assert_eq!(
214                partition_result, brute_result,
215                "Random test failed at iteration: threshold={threshold}, values={values:?}",
216            );
217
218            // Check that count of true elements is correct
219            assert_eq!(n, brute_result.iter().take(n).count());
220        }
221    }

223    #[test]
224    fn test_stability_dedicated() {
225        use rand::prelude::StdRng;
226        use rand::Rng;
227        use rand::SeedableRng;
228
229        let mut rng = StdRng::seed_from_u64(42);
230        for _ in 0..150 {
231            let len = rng.gen_range(0..50);
232            let values: Vec<u32> = (0..len).map(|_| rng.gen()).collect();
233            let threshold: u32 = rng.gen();
234
235            // Map values to their original indices for stability verification
236            let indices: Vec<usize> = (0..len).collect();
237            let tagged: Vec<(u32, usize)> = values
238                .iter()
239                .zip(indices.iter())
240                .map(|(&val, &idx)| (val, idx))
241                .collect();
242
243            let mut result = tagged.clone();
244            let n = partition(&mut result, |&(v, _)| v < threshold);
245
246            // Verify true part: original indices must be strictly increasing
247            for i in 0..n.saturating_sub(1) {
248                assert!(
249                    result[i].1 < result[i + 1].1,
250                    "True part not stable: result[{}].idx={}, result[{}].idx={}",
251                    i,
252                    result[i].1,
253                    i + 1,
254                    result[i + 1].1
255                );
256            }
257
258            // Verify false part: original indices must be strictly increasing
259            for i in n..result.len().saturating_sub(1) {
260                assert!(
261                    result[i].1 < result[i + 1].1,
262                    "False part not stable: result[{}].idx={}, result[{}].idx={}",
263                    i,
264                    result[i].1,
265                    i + 1,
266                    result[i + 1].1
267                );
268            }
269        }
270    }
