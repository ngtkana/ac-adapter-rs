54    // Updates: the proper ancestors of `x` (i.e. `x` should be already updated)
55    pub fn fix_red(&mut self, mut x: Ptr<T>) {
56        while color(*x.parent()) == Color::Red {
57            let mut p = x.parent().unwrap();
58
59            let Some(mut pp) = p.parent() else {
60                p.update();
61                *p.color() = Color::Black;
62                self.black_height += 1;
63                return;
64            };
65
66            // Case 1: `p` is the left child.
67            if *pp.left() == Some(p) {
68                // Case 1.1: `pp` is a 5-node.
69                if color(*pp.right()) == Color::Red {
70                    *p.color() = Color::Black;
71                    *pp.color() = Color::Red;
72                    *pp.right().unwrap().color() = Color::Black;
73                    p.update();
74                    pp.update();
75                    x = pp;
76                }
77                // Case 1.2: `pp` is a splayed 4-node.
78                else if *p.right() == Some(x) {
79                    rotate_left(p);
80                    rotate_right(pp);
81                    *x.color() = Color::Black;
82                    *pp.color() = Color::Red;
83                    p.update();
84                    pp.update();
85                    x.update();
86                    break;
87                }
88                // Case 1.3: `pp` is a straight 4-node.
89                else {
90                    rotate_right(pp);
91                    *p.color() = Color::Black;
92                    *pp.color() = Color::Red;
93                    pp.update();
94                    break;
95                }
96            }
97            // Case 2: `p` is the right child.
98            else {
99                // Case 2.1: `pp` is a 5-node.
100                if color(*pp.left()) == Color::Red {
101                    *p.color() = Color::Black;
102                    *pp.color() = Color::Red;
103                    *pp.left().unwrap().color() = Color::Black;
104                    p.update();
105                    pp.update();
106                    x = pp;
107                }
108                // Case 2.2: `pp` is a splayed 4-node.
109                else if *p.left() == Some(x) {
110                    rotate_right(p);
111                    rotate_left(pp);
112                    *x.color() = Color::Black;
113                    *pp.color() = Color::Red;
114                    p.update();
115                    pp.update();
116                    x.update();
117                    break;
118                }
119                // Case 2.3: `pp` is a straight 4-node.
120                else {
121                    rotate_left(pp);
122                    *p.color() = Color::Black;
123                    *pp.color() = Color::Red;
124                    pp.update();
125                    break;
126                }
127            }
128        }
129        self.root = Some(x.update_ancestors());
130    }

132    // Updates: the proper ancestors of `x` (i.e. `x` should be already updated)
133    pub fn fix_black(&mut self, black_violation: BlackViolation<T>) {
134        let BlackViolation { p, mut x } = black_violation;
135        loop {
136            if color(x) == Color::Red {
137                *x.unwrap().color() = Color::Black;
138                break;
139            }
140            let p = x.map_or(p, |mut x| *x.parent());
141            let Some(mut p) = p else {
142                self.black_height -= 1;
143                return;
144            };
145            // Case 1: `x` is the left child.
146            if *p.left() == x {
147                let mut s = p.right().unwrap();
148                if *s.color() == Color::Red {
149                    rotate_left(p);
150                    *p.color() = Color::Red;
151                    *s.color() = Color::Black;
152                    s = p.right().unwrap();
153                }
154                match (color(*s.left()), color(*s.right())) {
155                    // Case 1.1: `s` is a 2-node.
156                    (Color::Black, Color::Black) => {
157                        *s.color() = Color::Red;
158                        p.update();
159                        x = Some(p);
160                    }
161                    // Case 1.2: `s` is a left-leaning 3-node.
162                    (Color::Red, Color::Black) => {
163                        let mut c = s.left().unwrap();
164                        rotate_right(s);
165                        rotate_left(p);
166                        *c.color() = *p.color();
167                        *p.color() = Color::Black;
168                        s.update();
169                        break;
170                    }
171                    // Case 1.3: `s` is a right-leaning 3-node, or a 4-node.
172                    (_, Color::Red) => {
173                        rotate_left(p);
174                        *s.color() = *p.color();
175                        *p.color() = Color::Black;
176                        *s.right().unwrap().color() = Color::Black;
177                        break;
178                    }
179                }
180            }
181            // Case 2: `x` is the right child.
182            else {
183                let mut s = p.left().unwrap();
184                if *s.color() == Color::Red {
185                    rotate_right(p);
186                    *p.color() = Color::Red;
187                    *s.color() = Color::Black;
188                    s = p.left().unwrap();
189                }
190                match (color(*s.left()), color(*s.right())) {
191                    // Case 2.1: `s` is a 2-node.
192                    (Color::Black, Color::Black) => {
193                        *s.color() = Color::Red;
194                        p.update();
195                        x = Some(p);
196                    }
197                    // Case 2.2: `s` os a right-leaning 3-node.
198                    (Color::Black, Color::Red) => {
199                        let mut c = s.right().unwrap();
200                        rotate_left(s);
201                        rotate_right(p);
202                        *c.color() = *p.color();
203                        *p.color() = Color::Black;
204                        s.update();
205                        break;
206                    }
207                    // Case 2.3: `s` is a left-leaning 3-node, or a 4-node.
208                    (Color::Red, _) => {
209                        rotate_right(p);
210                        *s.color() = *p.color();
211                        *p.color() = Color::Black;
212                        *s.left().unwrap().color() = Color::Black;
213                        break;
214                    }
215                }
216            }
217        }
218        self.root = Some(match x {
219            None => {
220                p.unwrap().update();
221                p.unwrap().update_ancestors()
222            }
223            Some(x) => x.update_ancestors(),
224        });
225    }

227    pub fn transplant(&mut self, mut place: Ptr<T>, new: Option<Ptr<T>>) {
228        if let Some(mut p) = *place.parent() {
229            *(if *p.left() == Some(place) { p.left() } else { p.right() }) = new;
230        } else {
231            self.root = new;
232        }
233        if let Some(mut new) = new {
234            *new.parent() = *place.parent();
235        }
236    }

276fn rotate_left<T: Balance>(mut l: Ptr<T>) {
277    let p = *l.parent();
278    let mut r = l.right().unwrap();
279    let c = *r.left();
280
281    // Connect `p` and `r`.
282    *r.parent() = p;
283    if let Some(mut p) = p {
284        *(if *p.left() == Some(l) { p.left() } else { p.right() }) = Some(r);
285    }
286
287    // Connect `r` and `l`.
288    *l.parent() = Some(r);
289    *r.left() = Some(l);
290
291    // Connect `l` and `c`.
292    if let Some(mut c) = c {
293        *c.parent() = Some(l);
294    }
295    *l.right() = c;
296}
297
298fn rotate_right<T: Balance>(mut r: Ptr<T>) {
299    let p = *r.parent();
300    let mut l = r.left().unwrap();
301    let c = *l.right();
302
303    // Connect `p` and `l`.
304    *l.parent() = p;
305    if let Some(mut p) = p {
306        *(if *p.left() == Some(r) { p.left() } else { p.right() }) = Some(l);
307    }
308
309    // Connect `l` and `r`.
310    *r.parent() = Some(l);
311    *l.right() = Some(r);
312
313    // Connect `r` and `c`.
314    if let Some(mut c) = c {
315        *c.parent() = Some(r);
316    }
317    *r.left() = c;
318}

450        #[allow(clippy::too_many_lines)]
451        pub fn random(
452            rng: &mut StdRng,
453            mut new_node: impl Fn(&mut StdRng, Color) -> T,
454            black_height: u8,
455            red_vio: bool,
456            black_vio: bool,
457            force_black_leaf: bool,
458        ) -> (Self, Violations<T>) {
459            #[allow(clippy::too_many_lines)]
460            fn random_tree<T: Balance>(
461                rng: &mut StdRng,
462                new_node: &mut impl Fn(&mut StdRng, Color) -> T,
463                mut black_height: u8,
464                red_vio: bool,
465                black_vio: bool,
466                force_black_leaf: bool,
467                parent: Option<Ptr<T>>,
468            ) -> (Option<Ptr<T>>, Violations<T>) {
469                if black_height == 0 && force_black_leaf {
470                    return (None, Violations::default());
471                }
472                // Select the violation position here
473                let parent_color = parent.map_or(Color::Black, |mut p| *p.color());
474                let here_red_vio;
475                let here_black_vio;
476                let color;
477                match (red_vio, black_vio) {
478                    (false, false) => {
479                        here_red_vio = false;
480                        here_black_vio = false;
481                        color = match parent_color {
482                            Color::Red => Color::Black,
483                            Color::Black => {
484                                if rng.gen() {
485                                    Color::Red
486                                } else {
487                                    Color::Black
488                                }
489                            }
490                        }
491                    }
492                    (true, false) => {
493                        here_red_vio = parent_color == Color::Red
494                            && red_vio
495                            && rng.gen_ratio(1, 2u32.pow(u32::from(black_height)));
496                        here_black_vio = false;
497                        color = if here_red_vio {
498                            Color::Red
499                        } else {
500                            match parent_color {
501                                Color::Red => Color::Black,
502                                Color::Black => {
503                                    if (red_vio && black_height == 0) || rng.gen() {
504                                        Color::Red
505                                    } else {
506                                        Color::Black
507                                    }
508                                }
509                            }
510                        };
511                    }
512                    (false, true) => {
513                        here_red_vio = false;
514                        color = match parent_color {
515                            Color::Red => Color::Black,
516                            Color::Black => {
517                                if rng.gen() {
518                                    Color::Red
519                                } else {
520                                    Color::Black
521                                }
522                            }
523                        };
524                        here_black_vio = rng.gen_ratio(
525                            1,
526                            2u32.pow(u32::from(black_height) - u32::from(color == Color::Black)),
527                        );
528                    }
529                    (true, true) => panic!(),
530                }
531
532                if here_black_vio {
533                    black_height -= 1;
534                }
535
536                // Select the violation position at children
537                let mut left_red_vio = false;
538                let mut right_red_vio = false;
539                if red_vio && !here_red_vio {
540                    *(if rng.gen() { &mut left_red_vio } else { &mut right_red_vio }) = true;
541                }
542                let mut left_black_vio = false;
543                let mut right_black_vio = false;
544                if black_vio && !here_black_vio {
545                    *(if rng.gen() { &mut left_black_vio } else { &mut right_black_vio }) = true;
546                }
547
548                // Init vios
549                let here = if black_height == 0 && color == Color::Black {
550                    None
551                } else {
552                    Some(Ptr::new(new_node(rng, color)))
553                };
554                let mut vios = Violations::default();
555                if here_red_vio {
556                    vios.red_vios.push(here.unwrap());
557                }
558                if here_black_vio {
559                    vios.black_vios.push(BlackViolation { x: here, p: parent });
560                }
561
562                // Recurse
563                if let Some(mut here) = here {
564                    let children_h = black_height - u8::from(color == Color::Black);
565                    let (mut left, left_vio) = random_tree(
566                        rng,
567                        new_node,
568                        children_h,
569                        left_red_vio,
570                        left_black_vio,
571                        force_black_leaf,
572                        Some(here),
573                    );
574                    vios.append(left_vio);
575                    let (mut right, right_vio) = random_tree(
576                        rng,
577                        new_node,
578                        children_h,
579                        right_red_vio,
580                        right_black_vio,
581                        force_black_leaf,
582                        Some(here),
583                    );
584                    vios.append(right_vio);
585
586                    if let Some(left) = left.as_mut() {
587                        *left.parent() = Some(here);
588                    }
589                    if let Some(right) = right.as_mut() {
590                        *right.parent() = Some(here);
591                    }
592                    *here.left() = left;
593                    *here.right() = right;
594                }
595                (here, vios)
596            }
597            let (root, vios) = random_tree(
598                rng,
599                &mut new_node,
600                black_height,
601                red_vio,
602                black_vio,
603                force_black_leaf,
604                None,
605            );
606            (Tree { root, black_height }, vios)
607        }

609        pub fn validate(&self) {
610            fn validate<T: Balance>(x: Option<Ptr<T>>) -> Result<u8, String> {
611                let Some(mut x) = x else {
612                    return Ok(0);
613                };
614                let left_height = validate(*x.left())?;
615                let right_height = validate(*x.right())?;
616                (left_height == right_height
617                    || left_height == right_height + 1
618                    || left_height + 1 == right_height)
619                    .then_some(())
620                    .ok_or_else(|| {
621                        format!("The difference of black height is greater than 1: {x:?}")
622                    })?;
623
624                // Parent pointer incinsistency
625                if let Some(mut left) = *x.left() {
626                    let p = *left.parent();
627                    (p == Some(x)).then_some(()).ok_or_else(|| {
628                        format!(
629                            "parent pointer inconsistency between {x:?} and its left child {left:?}. \
630                             The parent of the left child is {p:?}",
631                        )
632                    })?;
633                }
634                if let Some(mut right) = *x.right() {
635                    let p = *right.parent();
636                    (p == Some(x)).then_some(()).ok_or_else(|| {
637                        format!(
638                            "parent pointer inconsistency between {x:?} and its right child {right:?}. \
639                             The parent of the right child is {p:?}",
640                        )
641                    })?;
642                }
643                Ok(left_height.max(right_height) + u8::from(*x.color() == Color::Black))
644            }
645            || -> Result<(), String> {
646                let h = validate(self.root)?;
647                (self.black_height == h || self.black_height == h + 1)
648                    .then_some(())
649                    .ok_or_else(|| format!("black height of {self} is not {h}"))?;
650                if let Some(mut x) = self.root {
651                    // Parent is None
652                    (x.parent().is_none())
653                        .then_some(())
654                        .ok_or_else(|| format!("parent of {x:?} is not None"))?;
655                }
656                Ok(())
657            }()
658            .unwrap_or_else(|e| panic!("Validation error: {e}\nTree: {self}"));
659        }

661        pub fn collect(&self) -> Vec<Ptr<T>> {
662            fn extend<T: Balance>(x: Option<Ptr<T>>, out: &mut Vec<Ptr<T>>) {
663                if let Some(mut x) = x {
664                    extend(*x.left(), out);
665                    out.push(x);
666                    extend(*x.right(), out);
667                }
668            }
669            let mut out = Vec::new();
670            extend(self.root, &mut out);
671            out
672        }

693    impl<T: Balance> fmt::Debug for Violations<T> {
694        fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
695            f.debug_struct("Violations")
696                .field("red_vios", &self.red_vios)
697                .field("black_vios", &self.black_vios)
698                .finish()
699        }
700    }
701    impl<T: Balance> Violations<T> {
702        pub fn append(&mut self, other: Self) {
703            self.red_vios.extend(other.red_vios);
704            self.black_vios.extend(other.black_vios);
705        }
706
707        pub fn collect(tree: &Tree<T>) -> Self {
708            fn extend<T: Balance>(
709                x: Option<Ptr<T>>,
710                p: Option<Ptr<T>>,
711                vios: &mut Violations<T>,
712            ) -> u8 {
713                let Some(mut x) = x else { return 0 };
714                let left_height = extend(*x.left(), Some(x), vios);
715                let right_height = extend(*x.right(), Some(x), vios);
716                match left_height.cmp(&right_height) {
717                    Ordering::Less => vios.black_vios.push(BlackViolation {
718                        p: Some(x),
719                        x: *x.left(),
720                    }),
721                    Ordering::Greater => vios.black_vios.push(BlackViolation {
722                        p: Some(x),
723                        x: *x.right(),
724                    }),
725                    Ordering::Equal => {}
726                }
727                if let Some(mut p) = p {
728                    if *p.color() == Color::Red && *x.color() == Color::Red {
729                        vios.red_vios.push(x);
730                    }
731                }
732                left_height.max(right_height) + u8::from(*x.color() == Color::Black)
733            }
734            let mut vios = Self::default();
735            let h = extend(tree.root, None, &mut vios);
736            if tree.black_height != h {
737                vios.black_vios.push(BlackViolation {
738                    p: None,
739                    x: tree.root,
740                });
741            }
742            vios
743        }
744    }

898    struct Node {
899        pub value: i32,

903        pub left: Option<Ptr<Self>>,
904        pub right: Option<Ptr<Self>>,

906    fn sum(p: Option<Ptr<Node>>) -> i32 {
907        p.map_or(0, |p| p.sum)
908    }
909    impl super::Balance for Node {
910        fn update(&mut self) {
911            self.sum = sum(self.left) + self.value + sum(self.right);
912        }
913
914        fn push(&mut self) {}
915
916        fn color(&mut self) -> &mut Color {
917            &mut self.color
918        }
919
920        fn parent(&mut self) -> &mut Option<Ptr<Self>> {
921            &mut self.parent
922        }
923
924        fn left(&mut self) -> &mut Option<Ptr<Self>> {
925            &mut self.left
926        }
927
928        fn right(&mut self) -> &mut Option<Ptr<Self>> {
929            &mut self.right
930        }
931    }
932
933    impl Tree<Node> {
934        fn random_sum(
935            rng: &mut StdRng,
936            black_height: u8,
937            red_vio: bool,
938            black_vio: bool,
939        ) -> (Self, Violations<Node>) {
940            fn new_node(rng: &mut StdRng, color: Color) -> Node {
941                let value = rng.gen_range(0..VALUE_LIM);
942                Node {
943                    value,
944                    sum: value,
945                    color,
946                    parent: None,
947                    left: None,
948                    right: None,
949                }
950            }
951            fn update(p: Option<Ptr<Node>>) {
952                if let Some(mut p) = p {
953                    update(p.left);
954                    update(p.right);
955                    p.update();
956                }
957            }
958
959            let (tree, vios) = Self::random(rng, new_node, black_height, red_vio, black_vio, false);
960            update(tree.root);
961            (tree, vios)
962        }
963
964        fn validate_sum(&self) {
965            fn validate_sum(p: Option<Ptr<Node>>) -> Result<(), String> {
966                if let Some(p) = p {
967                    validate_sum(p.left)?;
968                    let expected = sum(p.left) + p.value + sum(p.right);
969                    (p.sum == expected).then_some(()).ok_or_else(|| {
970                        format!(
971                            "Sum is incorrect at {:?}. Expected {}, but cached {}",
972                            p, expected, p.sum
973                        )
974                    })?;
975                    validate_sum(p.right)?;
976                }
977                Ok(())
978            }
979            validate_sum(self.root).unwrap_or_else(|e| {
980                panic!(
981                    "{e}:\n Tree: {}",
982                    self.format_by(|p| format!("<{:?}:{},{}>", p, p.value, p.sum))
983                )
984            });
985        }
986    }
