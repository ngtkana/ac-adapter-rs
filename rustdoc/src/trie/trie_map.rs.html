6/// A map base on a trie.
7#[derive(Clone, PartialEq)]
8pub struct TrieMap<V>(Option<Box<Node<V>>>);

10impl<V: Debug> Debug for TrieMap<V> {
11    fn fmt(&self, w: &mut Formatter<'_>) -> fmt::Result {
12        let mut f = w.debug_map();
13        self.for_each_kv(|k, v| {
14            f.key(&k.to_vec()).value(v);
15        });
16        f.finish()
17    }
18}

26impl<V> TrieMap<V> {
27    /// Makes a new empty TrieMap.
28    ///
29    /// Does not allocate anything on its own.
30    ///
31    /// # Examples
32    ///
33    /// Basic usage:
34    ///
35    /// ```
36    /// use std::iter::once;
37    /// use trie::TrieMap;
38    ///
39    /// let mut map = TrieMap::new();
40    ///
41    /// // entries can now be inserted into the empty map
42    /// map.insert(once(1), "a");
43    /// ```
44    pub fn new() -> Self {
45        Self(None)
46    }

48    /// Inserts a key-value pair into the map.
49    ///
50    /// If the map did not have this key present, `None` is returned.
51    ///
52    /// If the map did have this key present, the value is updated, and the old
53    /// value is returned. The key is not updated, though.
54    ///
55    /// # Examples
56    ///
57    /// Basic usage:
58    ///
59    /// ```
60    /// use std::iter::once;
61    /// use trie::TrieMap;
62    ///
63    /// let mut map = TrieMap::new();
64    /// assert_eq!(map.insert(once(17), "a"), None);
65    /// // assert_eq!(map.is_empty(), false); TODO: unimplemented
66    ///
67    /// map.insert(once(17), "b");
68    /// assert_eq!(map.insert(once(17), "c"), Some("b"));
69    /// // assert_eq!(map[&37], "c"); TODO: unimplemented
70    /// ```
71    pub fn insert(&mut self, key: impl IntoIterator<Item = usize>, value: V) -> Option<V> {
72        let mut key = key.into_iter();
73        let me = self.0.get_or_insert_with(|| Box::new(Node::new()));
74        match key.next() {
75            Some(next) => me.child[next].insert(key, value),
76            None => me.value.replace(value),
77        }
78    }

80    /// Removes a key from the map, returning the stored key and value if the key
81    /// was previously in the map.
82    ///
83    /// # Examples
84    ///
85    /// Basic usage:
86    ///
87    /// ```
88    /// use std::iter::once;
89    /// use trie::TrieMap;
90    ///
91    /// let mut map = TrieMap::new();
92    /// map.insert(once(1), "a");
93    /// assert_eq!(map.remove(once(1)), Some("a"));
94    /// assert_eq!(map.remove(once(1)), None);
95    /// ```
96    pub fn remove(&mut self, key: impl IntoIterator<Item = usize>) -> Option<V> {
97        let mut key = key.into_iter();
98        let me = self.0.as_deref_mut()?;
99        match key.next() {
100            Some(next) => me.child[next].remove(key),
101            // TODO:
102            // 不要になったノードを削除したいです。しかし、サイズを管理しておかないと実現できません。
103            None => me.value.take(),
104        }
105    }

158    /// Inserts a `value` at `key` if it is [`None`], then returns a mutable reference
159    /// to the contained value.
160    ///
161    /// # Examples
162    ///
163    /// ```
164    /// use trie::TrieMap;
165    /// use std::iter::once;
166    ///
167    /// let mut map = TrieMap::new();
168    /// map.insert(once(1), "a");
169    ///
170    /// // Existing
171    /// map.get_or_insert(once(1), "b");
172    /// assert_eq!(map.get(once(1)), Some(&"a"));
173    ///
174    /// // New
175    /// map.get_or_insert(once(2), "c");
176    /// assert_eq!(map.get(once(2)), Some(&"c"));
177    ///
178    /// // assert_eq!(map[&1], "b"); TODO: unimplemented
179    pub fn get_or_insert(&mut self, key: impl IntoIterator<Item = usize>, value: V) -> &mut V {
180        let mut key = key.into_iter();
181        let me = self.0.get_or_insert_with(|| Box::new(Node::new()));
182        match key.next() {
183            Some(next) => me.child[next].get_or_insert(key, value),
184            None => me.value.get_or_insert(value),
185        }

188    /// Inserts a value computed from `f` at `key` if it is [`None`],
189    /// then returns a mutable reference to the contained value.
190    ///
191    /// # Examples
192    ///
193    /// ```
194    /// use trie::TrieMap;
195    /// use std::iter::once;

197    /// let mut map = TrieMap::new();
198    /// map.insert(once(1), "a");

205    /// map.get_or_insert_with(once(2), || "c");
206    /// assert_eq!(map.get(once(2)), Some(&"c"));
207    ///
208    /// // assert_eq!(map[&1], "b"); TODO: unimplemented
209    pub fn get_or_insert_with(
210        &mut self,
211        key: impl IntoIterator<Item = usize>,
212        f: impl FnOnce() -> V,
213    ) -> &mut V {
214        let mut key = key.into_iter();
215        let me = self.0.get_or_insert_with(|| Box::new(Node::new()));
216        match key.next() {
217            Some(next) => me.child[next].get_or_insert_with(key, f),
218            None => me.value.get_or_insert_with(f),

222    /// Visits all the "existing" nodes corresponding to the preficies of the key.
223    ///
224    /// # Examples
225    ///
226    /// Basic usage:
227    ///
228    /// ```
229    /// use std::iter::once;
230    /// use trie::TrieMap;
231    ///
232    /// let mut map = TrieMap::new();
233    /// map.insert(vec![1], "a");
234    /// map.insert(vec![1, 1, 1], "c");
235    ///
236    /// // Corresponding an existing key.
237    /// let mut expected = [None, Some("a"), None, Some("c")].iter();
238    /// map.for_each_prefix(vec![1, 1, 1].into_iter(), |trie| {
239    ///     let expected = expected.next().unwrap().as_ref();
240    ///     assert_eq!(trie.get(None.into_iter()), expected);
241    /// });
242    ///
243    /// // No, but falls within the trie.
244    /// let mut expected = [None, Some("a"), None].iter();
245    /// map.for_each_prefix(vec![1, 1].into_iter(), |trie| {
246    ///     let expected = expected.next().unwrap().as_ref();
247    ///     assert_eq!(trie.get(None.into_iter()), expected);
248    /// });
249    ///
250    /// // Runs off thte trie.
251    /// let mut expected = [None, Some("a"), None, Some("c")].iter();
252    /// map.for_each_prefix(vec![1, 1, 1, 1].into_iter(), |trie| {
253    ///     let expected = expected.next().unwrap().as_ref();
254    ///     assert_eq!(trie.get(None.into_iter()), expected);
255    /// });
256    /// ```
257    pub fn for_each_prefix(
258        &self,
259        key: impl IntoIterator<Item = usize>,
260        mut visit: impl FnMut(&Self),
261    ) {
262        let mut key = key.into_iter();
263        if let Some(me) = self.0.as_deref() {
264            visit(self);
265            let next = key.next();
266            if let Some(next) = next {
267                me.child[next].for_each_prefix(key, visit);
268            }
269        }
270    }
271
272    /// Visits all the pairs of a key of a values in the trie, in lexicographic order.
273    ///
274    /// # Examples
275    ///
276    /// Basic usage:
277    ///
278    /// ```
279    /// use std::iter::once;
280    /// use trie::TrieMap;
281    ///
282    /// let mut map = TrieMap::new();
283    /// map.insert(vec![1, 2, 2], "a");
284    /// map.insert(vec![2], "b");
285    /// map.insert(vec![1], "c");
286    /// map.insert(vec![1, 1, 1], "d");
287    ///
288    /// let mut expected = vec![
289    ///     (vec![1], "c"),
290    ///     (vec![1, 1, 1], "d"),
291    ///     (vec![1, 2, 2], "a"),
292    ///     (vec![2], "b"),
293    /// ]
294    /// .into_iter();
295    /// map.for_each_kv(|k, &v| {
296    ///     let (ek, ev) = expected.next().unwrap();
297    ///     assert_eq!(k, ek.as_slice());
298    ///     assert_eq!(v, ev);
299    /// });
300    /// ```
301    pub fn for_each_kv(&self, mut visit: impl FnMut(&[usize], &V)) {
302        let mut prefix = Vec::new();
303        self.for_each_kv_impl(&mut prefix, &mut visit);
304        assert!(prefix.is_empty());
305    }
306
307    fn for_each_kv_impl(&self, prefix: &mut Vec<usize>, visit: &mut impl FnMut(&[usize], &V)) {
308        if let Some(me) = self.0.as_deref() {
309            if let Some(value) = me.value.as_ref() {
310                visit(prefix, value);
311            }
312            for (i, child) in me.child.iter().enumerate() {
313                prefix.push(i);
314                child.for_each_kv_impl(prefix, visit);
315                prefix.pop();
316            }
317        }
318    }

342    use super::TrieMap;
343    use rand::prelude::*;
344    use std::collections::BTreeMap;
345    use test_case::test_case;

347    #[allow(clippy::unused_unit)]
348    #[test_case(200, 2; "short")]
349    #[test_case(200, 10; "mid")]
350    #[test_case(200, 100; "long")]
351    fn test_trie_map_rand(iter: usize, len_max: usize) {
352        let mut rng = StdRng::seed_from_u64(42);
353        let mut trie = TrieMap::new();
354        let mut btree_map = BTreeMap::new();
355        for _ in 0..iter {
356            let n = rng.gen_range(1..=len_max);
357            let s = rand::distributions::Uniform::new(0, 26)
358                .sample_iter(&mut rng)
359                .take(n)
360                .collect::<Vec<_>>();
361
362            match rng.gen_range(0..4) {
363                // insert
364                0 => {
365                    let init_value = 0;
366                    let trie_exist = trie.insert(s.iter().copied(), init_value);
367                    let btree_map_exist = btree_map.insert(s.clone(), init_value);
368                    assert_eq!(trie_exist, btree_map_exist);
369                }
370                // remove
371                1 => {
372                    let trie_exist = trie.remove(s.iter().copied());
373                    let btree_map_exist = btree_map.remove(&s);
374                    assert_eq!(trie_exist, btree_map_exist);
375                }
376                // get
377                2 => {
378                    let trie_exist = trie.get(s.iter().copied());
379                    let btree_map_exist = btree_map.get(&s);
380                    assert_eq!(trie_exist, btree_map_exist);
381                }
382                // get_mut
383                3 => {
384                    let trie_value = trie.get_mut(s.iter().copied());
385                    let btree_map_value = btree_map.get_mut(&s);
386                    assert_eq!(trie_value, btree_map_value);
387                    if let Some(trie_value) = trie_value {
388                        *trie_value += 1;
389                    }
390                    if let Some(btree_map_value) = btree_map_value {
391                        *btree_map_value += 1;
392                    }
393                }
394                _ => unreachable!(),
395            }
396
397            println!("trie = {trie:?}");
398            println!("btree_map = {btree_map:?}");
399            println!();
400        }
401    }
