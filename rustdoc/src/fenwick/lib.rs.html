1//! A Fenwick tree (Binary Indexed Tree) for efficient range queries.
2//!
3//! This crate provides a generic implementation of Fenwick trees,
4//! which support point updates and prefix/range queries in O(log n) time.
5//!
6//! # When to use
7//!
8//! A Fenwick tree is useful when you need to:
9//! - Quickly update individual elements
10//! - Query aggregate values over ranges (sum, XOR, min, etc.)
11//! - Process online queries where the dataset changes dynamically
12//!
13//! It's particularly common in competitive programming for range sum queries.
14//!
15//! # Examples
16//!
17//! ```
18//! use fenwick::{Fenwick, Op, OpSub};
19//!
20//! struct Sum;
21//! impl Op for Sum {
22//!     type Value = i64;
23//!     fn identity() -> i64 { 0 }
24//!     fn add(a: &i64, b: &i64) -> i64 { a + b }
25//! }
26//! impl OpSub for Sum {
27//!     fn sub(a: &i64, b: &i64) -> i64 { a - b }
28//! }
29//!
30//! let mut tree = Fenwick::<Sum>::new(5);
31//! tree.add(1, &10);
32//! tree.add(3, &20);
33//!
34//! // Sum of elements in [1, 4)
35//! assert_eq!(tree.fold(1..4), 30);
36//! ```
37//!
38//! # How it works
39//!
40//! This crate uses a generic [`Op`] trait to define the aggregation operation.
41//! Your type must implement:
42//! - An identity element (via [`Op::identity`])
43//! - An associative binary operation (via [`Op::add`])
44//!
45//! For range queries over arbitrary intervals, additionally implement [`OpSub`]
46//! to support subtraction. This enables the formula:
47//! `range_sum(start..end) = prefix_sum(end) - prefix_sum(start)`.
48//!
49//! # Core Items
50//!
51//! - [`Fenwick<O>`]: The main data structure for efficient range queries
52//! - [`Op`]: Trait defining associative operations (identity + add)
53//! - [`OpSub`]: Extension of [`Op`] that also supports subtraction
54//!
55//! # Complexity
56//!
57//! - `new(n)`: O(n)
58//! - `add()`, `sub()`: O(log n)
59//! - `fold_to()`, `fold()`: O(log n)
60

67/// A trait for defining associative operations on values.
68///
69/// This trait encapsulates the core operations needed for a Fenwick tree
70/// to perform efficient range queries and updates. Implementations must
71/// provide an identity element and an associative binary operation.
72///
73/// # Safety and Correctness
74///
75/// For the Fenwick tree to work correctly, `add` must be associative:
76/// for all values `a`, `b`, `c`, the operation must satisfy associativity.
77///
78/// # Examples
79///
80/// ```
81/// # use fenwick::Op;
82/// struct AddOp;
83/// impl Op for AddOp {
84///     type Value = i64;
85///     fn identity() -> Self::Value { 0 }
86///     fn add(a: &Self::Value, b: &Self::Value) -> Self::Value {
87///         a + b
88///     }
89/// }
90/// ```
91pub trait Op {
92    /// The type of values this operation works with.
93    type Value;
94
95    /// Returns the identity element for this operation.
96    ///
97    /// This is the element that, when combined with any value `v` via
98    /// `add`, leaves `v` unchanged.
99    fn identity() -> Self::Value;
100
101    /// Computes the result of the associative operation: `a ⊕ b`.
102    fn add(a: &Self::Value, b: &Self::Value) -> Self::Value;
103}
104
105/// An extension of [`Op`] that also supports subtraction.
106///
107/// This trait is needed to compute range queries over an arbitrary range
108/// `[start, end)`. With only addition, you can only compute prefix sums.
109/// With subtraction, you can compute range sums via the formula:
110/// `range_sum(start..end) = prefix_sum(end) - prefix_sum(start)`.
111///
112/// # Examples
113///
114/// ```
115/// # use fenwick::OpSub;
116/// struct AddOp;
117/// impl OpSub for AddOp {
118///     fn sub(a: &i64, b: &i64) -> i64 {
119///         a - b
120///     }
121/// }
122/// # impl fenwick::Op for AddOp {
123/// #     type Value = i64;
124/// #     fn identity() -> i64 { 0 }
125/// #     fn add(a: &i64, b: &i64) -> i64 { a + b }
126/// # }
127/// ```
128pub trait OpSub: Op {
129    /// Computes the result of the inverse operation: `a ⊖ b`.
130    fn sub(a: &Self::Value, b: &Self::Value) -> Self::Value;
131}

133/// A Fenwick tree (also known as a Binary Indexed Tree).
134///
135/// A Fenwick tree is a data structure that efficiently supports:
136/// - **Point updates**: Update a single element in O(log n) time
137/// - **Prefix queries**: Compute the fold of elements in [0, i) in O(log n) time
138/// - **Range queries**: (with [`OpSub`]) Compute the fold of elements in [i, j) in O(log n) time
139///
140/// The tree uses a generic operation [`Op`] to define the aggregation function.
141///
142/// # Memory
143///
144/// A `Fenwick<O>` uses O(n) space where n is the length of the tree.
145///
146/// # Examples
147///
148/// Creating and updating a tree with addition:
149///
150/// ```
151/// # use fenwick::{Fenwick, Op};
152/// struct AddOp;
153/// impl Op for AddOp {
154///     type Value = i64;
155///     fn identity() -> i64 { 0 }
156///     fn add(a: &i64, b: &i64) -> i64 { a + b }
157/// }
158///
159/// let mut tree = Fenwick::<AddOp>::new(5);
160/// tree.add(2, &10);  // Add 10 at index 2
161/// tree.add(4, &5);   // Add 5 at index 4
162/// let sum = tree.fold_to(..5);  // Sum of [0, 5)
163/// assert_eq!(sum, 15);
164/// ```
165pub struct Fenwick<O: Op> {
166    items: Vec<O::Value>,
167    __maker: PhantomData<O>,
168}
169impl<T: Debug, O: Op<Value = T>> Debug for Fenwick<O> {
170    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
171        f.debug_map()
172            .entries(
173                (1usize..self.items.len()).map(|i| (i - (i & i.wrapping_neg())..i, &self.items[i])),
174            )
175            .finish()
176    }
177}

179impl<O: Op> Default for Fenwick<O> {
180    /// Creates a new empty Fenwick tree.
181    ///
182    /// # Examples
183    ///
184    /// ```
185    /// # use fenwick::Fenwick;
186    /// # struct AddOp;
187    /// # impl fenwick::Op for AddOp {
188    /// #     type Value = i64;
189    /// #     fn identity() -> i64 { 0 }
190    /// #     fn add(a: &i64, b: &i64) -> i64 { a + b }
191    /// # }
192    /// let tree: Fenwick<AddOp> = Default::default();
193    /// ```
194    fn default() -> Self {
195        Self {
196            items: vec![],
197            __maker: PhantomData,
198        }
199    }

201impl<T, O: Op<Value = T>> Fenwick<O> {
202    /// Creates a new Fenwick tree with the given length.
203    ///
204    /// All elements are initialized to the identity element of the operation.
205    ///
206    /// # Arguments
207    ///
208    /// * `len` - The number of logical elements in the tree
209    ///
210    /// # Examples
211    ///
212    /// ```
213    /// # use fenwick::{Fenwick, Op};
214    /// # struct AddOp;
215    /// # impl Op for AddOp {
216    /// #     type Value = i64;
217    /// #     fn identity() -> i64 { 0 }
218    /// #     fn add(a: &i64, b: &i64) -> i64 { a + b }
219    /// # }
220    /// let tree = Fenwick::<AddOp>::new(10);
221    /// ```
222    pub fn new(len: usize) -> Self
223    where
224        T: Clone,
225    {
226        Self {
227            items: vec![O::identity(); len + 1],
228            __maker: PhantomData,
229        }
230    }
231
232    /// Adds a value to the element at the given index.
233    ///
234    /// # Arguments
235    ///
236    /// * `index` - The index to update (must be < length)
237    /// * `value` - The value to add
238    ///
239    /// # Examples
240    ///
241    /// ```
242    /// # use fenwick::{Fenwick, Op};
243    /// # struct AddOp;
244    /// # impl Op for AddOp {
245    /// #     type Value = i64;
246    /// #     fn identity() -> i64 { 0 }
247    /// #     fn add(a: &i64, b: &i64) -> i64 { a + b }
248    /// # }
249    /// let mut tree = Fenwick::<AddOp>::new(5);
250    /// tree.add(2, &10);
251    /// ```
252    pub fn add(&mut self, mut index: usize, value: &T) {
253        index += 1;
254        while index < self.items.len() {
255            self.items[index] = O::add(&self.items[index], value);
256            index += index & index.wrapping_neg();
257        }
258    }
259
260    /// Computes the fold of all elements in the range `[0, end)`.
261    ///
262    /// # Arguments
263    ///
264    /// * `range` - A range `..end` specifying the prefix
265    ///
266    /// # Examples
267    ///
268    /// ```
269    /// # use fenwick::{Fenwick, Op};
270    /// # struct AddOp;
271    /// # impl Op for AddOp {
272    /// #     type Value = i64;
273    /// #     fn identity() -> i64 { 0 }
274    /// #     fn add(a: &i64, b: &i64) -> i64 { a + b }
275    /// # }
276    /// let mut tree = Fenwick::<AddOp>::new(5);
277    /// tree.add(0, &1);
278    /// tree.add(1, &2);
279    /// tree.add(2, &3);
280    /// assert_eq!(tree.fold_to(..3), 6);
281    /// ```
282    pub fn fold_to(&self, range: RangeTo<usize>) -> T {
283        let mut end = range.end;
284        let mut result = O::identity();
285        while end != 0 {
286            result = O::add(&result, &self.items[end]);
287            end -= end & end.wrapping_neg();
288        }
289        result
290    }
291}

293impl<T, O: OpSub<Value = T>> Fenwick<O> {
294    /// Subtracts a value from the element at the given index.
295    ///
296    /// This requires the operation to support subtraction (i.e., implement [`OpSub`]).
297    ///
298    /// # Arguments
299    ///
300    /// * `index` - The index to update (must be < length)
301    /// * `value` - The value to subtract
302    ///
303    /// # Examples
304    ///
305    /// ```
306    /// # use fenwick::{Fenwick, Op, OpSub};
307    /// # struct AddOp;
308    /// # impl Op for AddOp {
309    /// #     type Value = i64;
310    /// #     fn identity() -> i64 { 0 }
311    /// #     fn add(a: &i64, b: &i64) -> i64 { a + b }
312    /// # }
313    /// # impl OpSub for AddOp {
314    /// #     fn sub(a: &i64, b: &i64) -> i64 { a - b }
315    /// # }
316    /// let mut tree = Fenwick::<AddOp>::new(5);
317    /// tree.add(2, &10);
318    /// tree.sub(2, &3);
319    /// ```
320    pub fn sub(&mut self, mut index: usize, value: &T) {
321        index += 1;
322        while index < self.items.len() {
323            self.items[index] = O::sub(&self.items[index], value);
324            index += index & index.wrapping_neg();
325        }
326    }
327
328    /// Computes the fold of all elements in the range `[start, end)`.
329    ///
330    /// This requires the operation to support subtraction (i.e., implement [`OpSub`]).
331    ///
332    /// # Arguments
333    ///
334    /// * `range` - A range `[start, end)` specifying the interval
335    ///
336    /// # Examples
337    ///
338    /// ```
339    /// # use fenwick::{Fenwick, Op, OpSub};
340    /// # struct AddOp;
341    /// # impl Op for AddOp {
342    /// #     type Value = i64;
343    /// #     fn identity() -> i64 { 0 }
344    /// #     fn add(a: &i64, b: &i64) -> i64 { a + b }
345    /// # }
346    /// # impl OpSub for AddOp {
347    /// #     fn sub(a: &i64, b: &i64) -> i64 { a - b }
348    /// # }
349    /// let mut tree = Fenwick::<AddOp>::new(5);
350    /// tree.add(0, &1);
351    /// tree.add(1, &2);
352    /// tree.add(2, &3);
353    /// tree.add(3, &4);
354    /// assert_eq!(tree.fold(1..3), 5);  // 2 + 3
355    /// ```
356    pub fn fold(&self, range: Range<usize>) -> T {
357        let mut result = self.fold_to(..range.end);
358        result = O::sub(&result, &self.fold_to(..range.start));
359        result
360    }
