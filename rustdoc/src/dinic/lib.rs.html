1//! Solves maximum flow problem.
2//!
3//! # Basic usage
4//!
5//! 1. First, initialize [`Dinic`] with the number of vertices.
6//! 1. Insert edges with a method [`add_edge`](Dinic::add_edge).
7//! 1. Execute the algorithm by calling a method [`flow`](Dinic::flow) with the source `s` and sink `t`.
8//!
9//! ```
10//! use dinic::Dinic;
11//!
12//! let mut dinic = Dinic::new(3);
13//! dinic.add_edge(0, 1, 10);
14//! dinic.add_edge(1, 2, 15);
15//! dinic.add_edge(0, 2, 20);
16//!
17//! let flow = dinic.flow(0, 2);
18//! assert_eq!(flow, 30);
19//! ```
20//!
21//! # Restore the minimum cut
22//!
23//! If [`flow`](Dinic::flow) has called exactly once before, [`min_cut`](Dinic::min_cut) will
24//! return the minimum cut. [See the API document to detailed specs.](Dinic::min_cut)
25//!
26//! ```
27//! use dinic::Dinic;
28//!
29//! let mut dinic = Dinic::new(3);
30//! dinic.add_edge(0, 1, 10);
31//! dinic.add_edge(1, 2, 15);
32//! dinic.add_edge(0, 2, 20);
33//! dinic.flow(0, 2);
34//!
35//! assert_eq!(dinic.min_cut(0).as_slice(), &[true, false, false]);
36//! ```
37//!
38//!
39//! # Get the state of an edge or a vertex
40//!
41//! You can query the state of an edge via [`get_edge`](`Dinic::get_edge`). An [`EdgeKey`] object returned
42//! by [`add_edge`](Dinic::add_edge`) is necessary to query it.
43//!
44//! ```
45//! use dinic::Dinic;
46//! use dinic::Edge;
47//!
48//! let mut dinic = Dinic::new(3);
49//! dinic.add_edge(0, 1, 10);
50//! let key = dinic.add_edge(1, 2, 15);
51//! dinic.add_edge(0, 2, 20);
52//! dinic.flow(0, 2);
53//!
54//! assert_eq!(dinic.get_edge(key), Edge {
55//!     from: 1,
56//!     to: 2,
57//!     cap: 15,
58//!     flow: 10
59//! });
60//! ```
61//!
62//! Moreover [`get_edges`](Dinic::get_edges), [`get_network`](Dinic::get_network) will summarize the
63//! whole network.
64//!
65//! ```
66//! use dinic::Dinic;
67//! use dinic::Edge;
68//!
69//! let mut dinic = Dinic::new(3);
70//! dinic.add_edge(0, 1, 10);
71//! let key = dinic.add_edge(1, 2, 15);
72//! dinic.add_edge(0, 2, 20);
73//! dinic.flow(0, 2);
74//!
75//! let edges = dinic.get_edges();
76//! let network = dinic.get_network();
77//!
78//! // 0th edge
79//! assert_eq!(network[0][0], edges[0]);
80//! assert_eq!(edges[0], Edge {
81//!     from: 0,
82//!     to: 1,
83//!     cap: 10,
84//!     flow: 10
85//! });
86//!
87//! // 1st edge
88//! assert_eq!(network[1][0], edges[1]);
89//! assert_eq!(edges[1], Edge {
90//!     from: 1,
91//!     to: 2,
92//!     cap: 15,
93//!     flow: 10
94//! });
95//!
96//! // 2nd edge
97//! assert_eq!(network[0][1], edges[2]);
98//! assert_eq!(edges[2], Edge {
99//!     from: 0,
100//!     to: 2,
101//!     cap: 20,
102//!     flow: 20
103//! });
104//! ```
105//!
106//! You also can get the excess of vertices, but we do not provide an interface to get the excess of *a vertex* because
107//! it will take O ( m ) time.
108//!
109//! ```
110//! use dinic::Dinic;
111//! use dinic::Edge;
112//!
113//! let mut dinic = Dinic::new(3);
114//! dinic.add_edge(0, 1, 10);
115//! let key = dinic.add_edge(1, 2, 15);
116//! dinic.add_edge(0, 2, 20);
117//! dinic.flow(0, 2);
118//!
119//! assert_eq!(dinic.get_excess().as_slice(), &[-30, 0, 30]);
120//! ```
121//!
122//! If you use an unsigned type, [`get_excess`](Dinic::get_excess) will surely overflow because the
123//! excess of the source is almost always negative.
124//!
125//! ```should_panic
126//! use dinic::Dinic;
127//! use dinic::Edge;
128//!
129//! let mut dinic = Dinic::<u32>::new(2); // Force to use `u32` instead of `i32`.
130//! dinic.add_edge(0, 1, 10);
131//! dinic.flow(0, 1);
132//!
133//! dinic.get_excess(); // panics
134//! ```
135//!
136//! # Call `flow` more than once
137//!
138//! You can call [`flow`](Dinic::flow) more than once. `flow(s, t)` will augment the flow from `s`
139//! to `t` as much as possible. If [`flow`](Dinic::flow) is called with different `s` or `t` from
140//! the previous ones, it may yield non-zero excess at more than two points.
141//!
142//! ```
143//! use dinic::Dinic;
144//!
145//! let mut dinic = Dinic::new(3);
146//! dinic.add_edge(0, 1, 10);
147//! dinic.add_edge(1, 2, 15);
148//! dinic.add_edge(0, 2, 20);
149//!
150//! dinic.flow(0, 2);
151//! let aug = dinic.flow(1, 2);
152//! assert_eq!(aug, 5);
153//! assert_eq!(dinic.get_excess().as_slice(), &[-30, -5, 35]);
154//! ```
155//!
156//! # Change the capacity or the amount of flow. (dangerous operation)
157//!
158//! [`change_edge`](Dinic::change_edge) changes the capacity and the amount of flow of an edge.
159//! While it is so dangerous operation (safe variant wanted!), it is sometimes useful, for example,
160//! when solve a *maximum flow problem with lower limit*. [See the API document for detailed
161//! specs.](Dinic::change_edge)

173/// An adapter trait of the capacity.
174///
175/// This trait is implemented for all the integer types.
176pub trait Value:
177    Copy + Ord + Debug + Add<Output = Self> + AddAssign + Sub<Output = Self> + SubAssign + Sum
178{
179    /// Returns the zero.
180    fn zero() -> Self;
181    /// Returns the max value of `Self`.
182    fn infinity() -> Self;
183}

220    /// Inserts a new edge to the network.
221    ///
222    /// # Constraints
223    ///
224    /// - `from, to < n`
225    /// - `T::zero() <= cap`
226    ///
227    /// # Complexity
228    ///
229    /// O ( 1 ) amortized.
230    ///
231    /// # Examples
232    ///
233    /// ```
234    /// use dinic::Dinic;
235    ///
236    /// let mut dinic = Dinic::new(3);
237    /// dinic.add_edge(0, 1, 10);
238    /// dinic.add_edge(1, 2, 15);
239    /// dinic.add_edge(0, 2, 20);
240    ///
241    /// let flow = dinic.flow(0, 2);
242    /// assert_eq!(flow, 30);
243    /// ```
244    pub fn add_edge(&mut self, from: usize, to: usize, cap: T) -> EdgeKey {
245        assert!(
246            from < self.res.len() || to < self.res.len(),
247            "`Dinic::add_edge` is called with from = {}, to = {}, but the number of verticies is \
248             {}",
249            from,
250            to,
251            self.res.len()
252        );
253        assert!(
254            T::zero() <= cap,
255            "`Dinic::add_edge` is called with a negative `cap`"
256        );
257        let size_from = self.res[from].len();
258        let size_to = if from == to { self.res[to].len() + 1 } else { self.res[to].len() };
259        let edge_key = self.pos.len();
260        self.pos.push(__EdgeIndexer {
261            from,
262            index: size_from,
263        });
264        self.res[from].push(__ResidualEdge {
265            to,
266            cap,
267            rev: size_to,
268        });
269        self.res[to].push(__ResidualEdge {
270            to: from,
271            cap: T::zero(),
272            rev: size_from,
273        });
274        EdgeKey(edge_key)
275    }

286    /// - The answer should be in `T`.
287    ///
288    ///
289    /// # Complexity

307    pub fn flow(&mut self, s: usize, t: usize) -> T {
308        assert!(
309            s < self.res.len() && t < self.res.len(),
310            "`Dinic::flow` is called with `s` = {}, `t` = {}, while the number of vertices is {}",
311            s,
312            t,
313            self.res.len()
314        );
315        dinic_impl(&mut self.res, s, t, T::infinity())
316    }

360    /// Returns a vector of length `n`, such that the `i`-th element is `true` if and only if there
361    /// is a directed path from `s` to `i` in the residual network. The returned vector correponds
362    /// to a `s -- t` minimum cut after calling `self.flow(s, t)` exactly once.
363    ///
364    /// # Examples
365    ///
366    /// ```
367    /// use dinic::Dinic;
368    ///
369    /// let mut dinic = Dinic::new(3);
370    /// dinic.add_edge(0, 1, 10);
371    /// dinic.add_edge(1, 2, 15);
372    /// dinic.add_edge(0, 2, 20);
373    ///
374    /// dinic.flow(0, 2);
375    /// assert_eq!(dinic.min_cut(0).as_slice(), &[true, false, false]);
376    /// ```
377    pub fn min_cut(&self, s: usize) -> Vec<bool> {
378        let mut visited = vec![false; self.res.len()];
379        let mut queue = VecDeque::from(vec![s]);
380        while let Some(from) = queue.pop_front() {
381            visited[from] = true;
382            queue.extend(
383                self.res[from]
384                    .iter()
385                    .copied()
386                    .filter(|&__ResidualEdge { to, cap, .. }| {
387                        cap != T::zero() && !std::mem::replace(&mut visited[to], true)
388                    })
389                    .map(|__ResidualEdge { to, .. }| to),
390            );
391        }
392        visited
393    }

418    pub fn get_edge(&self, edge_key: EdgeKey) -> Edge<T> {
419        let EdgeKey(edge_key) = edge_key;
420        assert!(
421            edge_key < self.pos.len(),

476    /// Collects all the edges and arrange it in adjacent-list style.
477    ///
478    /// In each row, edges are sorted in order of addition.
479    ///
480    /// # Complexity
481    ///
482    /// O ( n + m )
483    ///
484    /// # Examples
485    ///
486    /// ```
487    /// use dinic::Dinic;
488    /// use dinic::Edge;
489    ///
490    /// let mut dinic = Dinic::new(3);
491    /// dinic.add_edge(0, 1, 10);
492    /// dinic.add_edge(1, 2, 15);
493    /// dinic.add_edge(0, 2, 20);
494    /// dinic.flow(0, 2);
495    ///
496    /// let network = dinic.get_network();
497    /// assert_eq!(network[0][0], Edge {
498    ///     from: 0,
499    ///     to: 1,
500    ///     cap: 10,
501    ///     flow: 10
502    /// });
503    /// assert_eq!(network[0][1], Edge {
504    ///     from: 0,
505    ///     to: 2,
506    ///     cap: 20,
507    ///     flow: 20
508    /// });
509    /// assert_eq!(network[1][0], Edge {
510    ///     from: 1,
511    ///     to: 2,
512    ///     cap: 15,
513    ///     flow: 10
514    /// });
515    /// ```
516    pub fn get_network(&self) -> Vec<Vec<Edge<T>>> {
517        let mut network = vec![Vec::new(); self.res.len()];
518        self.pos
519            .iter()
520            .map(|&edge_indexer| self.restore_edge(edge_indexer))
521            .for_each(|edge| network[edge.from].push(edge));
522        network
523    }

525    /// Returens the `Vec` of excess of all the vertices.
526    ///
527    /// `i`-th entry is the excess of vertex `i`.
528    ///
529    /// # Complexity
530    ///
531    /// O ( n + m )
532    ///
533    /// # Examples
534    ///
535    /// ```
536    /// use dinic::Dinic;
537    ///
538    /// let mut dinic = Dinic::new(3);
539    /// dinic.add_edge(0, 1, 10);
540    /// dinic.add_edge(1, 2, 15);
541    /// dinic.add_edge(0, 2, 20);
542    /// dinic.flow(0, 2);
543    ///
544    /// let excess = dinic.get_excess();
545    /// assert_eq!(excess.as_slice(), &[-30, 0, 30]);
546    /// ```
547    pub fn get_excess(&self) -> Vec<T> {
548        let mut excess = vec![T::zero(); self.res.len()];
549        self.pos
550            .iter()
551            .map(|&edge_indexer| self.restore_edge(edge_indexer))
552            .for_each(|Edge { to, flow, .. }| excess[to] += flow);
553        self.pos
554            .iter()
555            .map(|&edge_indexer| self.restore_edge(edge_indexer))
556            .for_each(|Edge { from, flow, .. }| excess[from] -= flow);
557        excess
558    }

560    /// For internal use.
561    ///
562    /// # Constraints
563    ///
564    /// `edge_indexer` is taken from `self.pos`
565    fn restore_edge(&self, edge_indexer: __EdgeIndexer) -> Edge<T> {
566        let __EdgeIndexer { from, index } = edge_indexer;
567        let __ResidualEdge { to, cap, rev } = self.res[from][index];
568        let rev = self.res[to][rev];
569        Edge {
570            from,
571            to,
572            cap: cap + rev.cap,
573            flow: rev.cap,
574        }
575    }

577    /// Changes the capacity and the amount of the edge corresponding to `edge_key` to `new_cap` and
578    /// `new_flow`, respectively. It does not change the capacity or the flow amount of other
579    /// edges. [See the module level documentation.](self)
580    ///
581    /// # Constraints
582    ///
583    /// - `T::zero() <= new_flow <= new_cap`
584    ///
585    /// # Complexity
586    ///
587    /// O ( 1 )
588    ///
589    /// # Examples
590    ///
591    /// Let us consider a bipartite matching problem on the graph `K _ 2`.
592    /// It can be solved as a maximum flow problem.
593    ///
594    /// ```
595    /// use dinic::Dinic;
596    ///
597    /// let mut dinic = Dinic::new(6);
598    /// let edge_0 = dinic.add_edge(0, 1, 1);
599    /// let edge_1 = dinic.add_edge(0, 2, 1);
600    /// let edge_2 = dinic.add_edge(1, 3, 1);
601    /// let edge_3 = dinic.add_edge(1, 4, 1);
602    /// let edge_4 = dinic.add_edge(2, 3, 1);
603    /// let edge_5 = dinic.add_edge(2, 4, 1);
604    /// let edge_6 = dinic.add_edge(3, 5, 1);
605    /// let edge_7 = dinic.add_edge(4, 5, 1);
606    ///
607    /// let flow = dinic.flow(0, 5);
608    /// assert_eq!(flow, 2);
609    ///
610    /// # // Actually, the "parallel" ones are the matching edges.
611    /// # assert_eq!(dinic.get_edge(edge_2).flow, 1);
612    /// # assert_eq!(dinic.get_edge(edge_3).flow, 0);
613    /// # assert_eq!(dinic.get_edge(edge_4).flow, 0);
614    /// # assert_eq!(dinic.get_edge(edge_5).flow, 1);
615    /// ```
616    ///
617    /// Now, let us add another constraint. Forbid to match the top ones.
618    /// In order to do this, we deminish the flow by `1` from the source to the sink, along this matching edge,
619    /// and diminish the capacity of this matching edge to `0`, so that the maching is of size `1`
620    /// and the `dinic` is also a feasible flow of value `1`.
621    ///
622    /// Now call [`Dinic::flow`] again with the same source and the same sink. So new matching will be of size `2` shaping like "X". So [`Dinic::flow`] will return the delta `1 = 2 - 1` of the flow.
623    ///
624    /// ```
625    /// # use dinic::Dinic;
626    /// #
627    /// # let mut dinic = Dinic::new(6);
628    /// # let edge_0 = dinic.add_edge(0, 1, 1);
629    /// # let edge_1 = dinic.add_edge(0, 2, 1);
630    /// # let edge_2 = dinic.add_edge(1, 3, 1);
631    /// # let edge_3 = dinic.add_edge(1, 4, 1);
632    /// # let edge_4 = dinic.add_edge(2, 3, 1);
633    /// # let edge_5 = dinic.add_edge(2, 4, 1);
634    /// # let edge_6 = dinic.add_edge(3, 5, 1);
635    /// # let edge_7 = dinic.add_edge(4, 5, 1);
636    ///
637    /// # let flow = dinic.flow(0, 5);
638    /// # assert_eq!(flow, 2);
639    /// #
640    /// # // Actually, the "parallel" ones are the matching edges.
641    /// # assert_eq!(dinic.get_edge(edge_2).flow, 1);
642    /// # assert_eq!(dinic.get_edge(edge_3).flow, 0);
643    /// # assert_eq!(dinic.get_edge(edge_4).flow, 0);
644    /// # assert_eq!(dinic.get_edge(edge_5).flow, 1);
645    /// #
646    /// dinic.change_edge(edge_0, 1, 0); // the edge from the source
647    /// dinic.change_edge(edge_2, 0, 0); // the matching edge
648    /// dinic.change_edge(edge_6, 1, 0); // the edge to the sink
649    ///
650    /// // now, `dinic` has a feasible flow of value `1`
651    ///
652    /// let augment = dinic.flow(0, 5);
653    /// assert_eq!(augment, 1); // and augmented by `1` and became `2`.
654    /// #
655    /// # assert_eq!(dinic.get_edge(edge_2).flow, 0);
656    /// # assert_eq!(dinic.get_edge(edge_3).flow, 1);
657    /// # assert_eq!(dinic.get_edge(edge_4).flow, 1);
658    /// # assert_eq!(dinic.get_edge(edge_5).flow, 0);
659    /// ```
660    ///
661    pub fn change_edge(&mut self, edge_key: EdgeKey, new_cap: T, new_flow: T) {
662        let EdgeKey(edge_key) = edge_key;
663        assert!(
664            edge_key < self.pos.len(),
665            "Called `Dinic::get_edge` with `edge_key` = {:?}, but the length of `Dinic::pos` is {}",
666            edge_key,
667            self.pos.len()
668        );
669        assert!(
670            T::zero() <= new_flow && new_flow <= new_cap,
671            "Called `Dinic::change_edge` by new_flow = {new_flow:?}, new_cap = {new_cap:?}"
672        );
673        let __EdgeIndexer { from, index } = self.pos[edge_key];
674        let __ResidualEdge { to, rev, cap } = &mut self.res[from][index];
675        *cap = new_cap - new_flow;
676        let to = *to;
677        let rev = *rev;
678        self.res[to][rev].cap = new_flow;
679    }
680}

683        write!(w, "{:?}", self.get_network())

723fn dinic_impl<T>(res: &mut [Vec<__ResidualEdge<T>>], s: usize, t: usize, flow_limit: T) -> T
724where
725    T: Value,
726{
727    assert_ne!(s, t);
728
729    let mut flow = T::zero();
730
731    loop {
732        // calculate labels
733        let mut label = vec![u32::MAX; res.len()];
734        label[s] = 0;
735        let mut queue = VecDeque::from(vec![s]);
736        while let Some(from) = queue.pop_front() {
737            for &__ResidualEdge { to, cap, .. } in &res[from] {
738                if cap == T::zero() || label[to] != u32::MAX {
739                    continue;
740                }
741                label[to] = label[from] + 1;
742                queue.push_back(to);
743            }
744        }
745
746        if label[t] == u32::MAX {
747            // saturated
748            return flow;
749        }
750
751        // make a current-dege data structure
752        let mut cur = vec![0; res.len()];
753        cur[t] = res[t].len();
754
755        // find augmenting paths
756        'PRIMAL_STEP: loop {
757            let mut path = Vec::<(usize, usize)>::new();
758            // depth-first search
759            'FIND_AUGUMENTING_PATH: loop {
760                let from = path.last().map_or(s, |&(x, i)| res[x][i].to);
761                loop {
762                    if let Some(&__ResidualEdge { to, cap, .. }) = res[from].get(cur[from]) {
763                        if cap == T::zero() || label[from] + 1 != label[to] {
764                            cur[from] += 1;
765                        } else {
766                            path.push((from, cur[from]));
767                            break;
768                        }
769                    } else if from == s {
770                        break 'PRIMAL_STEP;
771                    } else if from == t {
772                        break 'FIND_AUGUMENTING_PATH;
773                    } else {
774                        path.pop().unwrap();
775                        let from = path.last().map_or(s, |&(x, i)| res[x][i].to);
776                        cur[from] += 1;
777                        break;
778                    }
779                }
780            }
781
782            // augment the flow
783            let aug = path
784                .iter()
785                .map(|&(x, i)| res[x][i].cap)
786                .min()
787                .unwrap()
788                .min(flow_limit - flow);
789            if aug == T::zero() {
790                return flow;
791            }
792            flow += aug;
793            for (from, i) in path {
794                let __ResidualEdge { to, rev, .. } = res[from][i];
795                res[from][i].cap -= aug;
796                res[to][rev].cap += aug;
797            }
798        }
799    }
800}

897
