1//! # LazySegtree
2//!
3//! Defines a struct [`LazySegtree`] and a trait [`Op`] for a lazy segment tree.
4//!
5//! # Note
6//!
7//! It does not support binary searches.
8//!
9//! # Example
10//!
11//! ```
12//! use lazy_segtree::LazySegtree;
13//! use lazy_segtree::Op;
14//!
15//! enum O {}
16//! impl Op for O {
17//!     type Operator = u64;
18//!     type Value = u64;
19//!
20//!     fn identity() -> Self::Value {
21//!         0
22//!     }
23//!
24//!     fn op(lhs: &Self::Value, rhs: &Self::Value) -> Self::Value {
25//!         (*lhs).max(*rhs)
26//!     }
27//!
28//!     fn apply(op: &Self::Operator, value: &Self::Value) -> Self::Value {
29//!         *op + *value
30//!     }
31//!
32//!     fn identity_op() -> Self::Operator {
33//!         0
34//!     }
35//!
36//!     fn compose(op: &Self::Operator, other: &Self::Operator) -> Self::Operator {
37//!         *op + *other
38//!     }
39//! }
40//!
41//! let mut seg = LazySegtree::<O>::new(&[3, 1, 4, 1, 5, 9, 2, 6]);
42//! assert_eq!(seg.fold(0..8), 9);
43//! seg.range_apply(3..6, &2);
44//! assert_eq!(seg.fold(0..8), 11);
45//! ```
46use std::iter::FromIterator;
47use std::mem::replace;
48use std::ops::RangeBounds;
49
50/// Opertions for a lazy segment tree.
51pub trait Op {
52    /// The value type.
53    type Value;
54    /// The operator type.
55    type Operator: PartialEq;
56
57    /// Returns the identity value.
58    fn identity() -> Self::Value;
59    /// Multiplies two values.
60    fn op(lhs: &Self::Value, rhs: &Self::Value) -> Self::Value;
61    /// Applies an operator to a value.
62    fn apply(op: &Self::Operator, value: &Self::Value) -> Self::Value;
63    /// Returns the identity operator.
64    fn identity_op() -> Self::Operator;
65    /// Composes two operators.
66    /// The result of `compose(a, b)` is equivalent to applying `b` and then `a`.
67    fn compose(op: &Self::Operator, other: &Self::Operator) -> Self::Operator;
68}
69
70/// A lazy segment tree.
71#[derive(Debug, Clone)]
72pub struct LazySegtree<O: Op> {
73    values: Vec<O::Value>,
74    operators: Vec<O::Operator>,
75}
76impl<O: Op> LazySegtree<O> {
77    /// Constructs a new lazy segment tree.
78    /// You can use `from_iter` instead of this method.
79    pub fn new(values: &[O::Value]) -> Self
80    where
81        O::Value: Clone,
82        O::Operator: Clone,
83    {
84        let values_ = values;
85        let n = values_.len();
86        let mut values = vec![O::identity(); 2 * n];
87        values[n..].clone_from_slice(values_);
88        for i in (1..n).rev() {
89            values[i] = O::op(&values[i * 2], &values[i * 2 + 1]);
90        }
91        Self {
92            values,
93            operators: vec![O::identity_op(); 2 * n],
94        }
95    }
96
97    /// Applies an operator to a range.
98    pub fn range_apply<R: RangeBounds<usize>>(&mut self, range: R, f: &O::Operator) {
99        let n = self.operators.len() / 2;
100        let (l, r) = open(range, n);
101        if l == r {
102            return;
103        }
104        let l = n + l;
105        let r = n + r;
106        for p in (0..usize::BITS - r.leading_zeros()).rev() {
107            self.push(l >> p);
108            self.push((r - 1) >> p);
109        }
110        {
111            let mut l = l;
112            let mut r = r;
113            while l < r {
114                if l & 1 != 0 {
115                    self.operators[l] = O::compose(f, &self.operators[l]);
116                    l += 1;
117                }
118                if r & 1 != 0 {
119                    r -= 1;
120                    self.operators[r] = O::compose(f, &self.operators[r]);
121                }
122                l >>= 1;
123                r >>= 1;
124            }
125        }
126        for p in 1..usize::BITS - r.leading_zeros() {
127            self.update(l >> p);
128            self.update((r - 1) >> p);
129        }
130    }
131
132    /// Folds a range.
133    pub fn fold<R: RangeBounds<usize>>(&mut self, range: R) -> O::Value {
134        let n = self.operators.len() / 2;
135        let (mut l, mut r) = open(range, n);
136        if l == r {
137            return O::identity();
138        }
139        l += n;
140        r += n;
141        for p in (0..usize::BITS - r.leading_zeros()).rev() {
142            self.push(l >> p);
143            self.push((r - 1) >> p);
144        }
145        for p in 1..usize::BITS - r.leading_zeros() {
146            self.update(l >> p);
147            self.update((r - 1) >> p);
148        }
149        let mut left = O::identity();
150        let mut right = O::identity();
151        while l < r {
152            if l & 1 != 0 {
153                left = O::op(&left, &O::apply(&self.operators[l], &self.values[l]));
154                l += 1;
155            }
156            if r & 1 != 0 {
157                r -= 1;
158                right = O::op(&O::apply(&self.operators[r], &self.values[r]), &right);
159            }
160            l >>= 1;
161            r >>= 1;
162        }
163        O::op(&left, &right)
164    }
165
166    /// Returns the value at the index `i`.
167    pub fn get(&self, i: usize) -> O::Value
168    where
169        O::Value: Clone,
170    {
171        let mut i = self.operators.len() / 2 + i;
172        let mut value = self.values[i].clone();
173        while i > 0 {
174            value = O::apply(&self.operators[i], &value);
175            i >>= 1;
176        }
177        value
178    }
179
180    /// Returns the values as a vector.
181    /// It takes $O(n \log n)$ time because it calls `get` for each index.
182    pub fn collect(&self) -> Vec<O::Value>
183    where
184        O::Value: Clone,
185    {
186        (0..self.operators.len() / 2).map(|i| self.get(i)).collect()
187    }
188
189    fn push(&mut self, i: usize) {
190        let a = replace(&mut self.operators[i], O::identity_op());
191        self.values[i] = O::apply(&a, &self.values[i]);
192        if i < self.operators.len() / 2 {
193            self.operators[i << 1] = O::compose(&a, &self.operators[i << 1]);
194            self.operators[i << 1 | 1] = O::compose(&a, &self.operators[i << 1 | 1]);
195        }
196    }
197
198    fn update(&mut self, i: usize) {
199        self.values[i] = O::op(
200            &O::apply(&self.operators[i << 1], &self.values[i << 1]),
201            &O::apply(&self.operators[i << 1 | 1], &self.values[i << 1 | 1]),
202        );
203    }
204}
205
206impl<O: Op> FromIterator<O::Value> for LazySegtree<O>
207where
208    O::Value: Clone,
209    O::Operator: Clone,
210{
211    fn from_iter<T: IntoIterator<Item = O::Value>>(iter: T) -> Self {
212        Self::new(&iter.into_iter().collect::<Vec<_>>())
213    }
214}
215
216fn open<B: RangeBounds<usize>>(bounds: B, n: usize) -> (usize, usize) {
217    use std::ops::Bound;
218    let start = match bounds.start_bound() {
219        Bound::Unbounded => 0,
220        Bound::Included(&x) => x,
221        Bound::Excluded(&x) => x + 1,
222    };
223    let end = match bounds.end_bound() {
224        Bound::Unbounded => n,
225        Bound::Included(&x) => x + 1,
226        Bound::Excluded(&x) => x,
227    };
228    (start, end)
229}

231#[cfg(test)]
232mod tests {
233    use super::*;
234    use rand::rngs::StdRng;
235    use rand::Rng;
236    use rand::SeedableRng;

238    const P: u64 = 998_244_353;
239
240    #[derive(Debug, Clone, Copy, PartialEq)]
241    struct Affine {
242        a: u64,
243        b: u64,
244    }
245    #[derive(Debug, Clone, Copy, PartialEq)]
246    struct Value {
247        value: u64,
248        len: usize,
249    }
250    enum O {}
251    impl Op for O {
252        type Operator = Affine;
253        type Value = Value;
254
255        fn identity() -> Self::Value {
256            Value { value: 0, len: 0 }
257        }
258
259        fn op(lhs: &Self::Value, rhs: &Self::Value) -> Self::Value {
260            Value {
261                value: (lhs.value + rhs.value) % P,
262                len: lhs.len + rhs.len,
263            }
264        }
265
266        fn apply(op: &Self::Operator, value: &Self::Value) -> Self::Value {
267            Value {
268                value: (op.a * value.value + op.b * value.len as u64) % P,
269                len: value.len,
270            }
271        }
272
273        fn identity_op() -> Self::Operator {
274            Affine { a: 1, b: 0 }
275        }
276
277        fn compose(op: &Self::Operator, other: &Self::Operator) -> Self::Operator {
278            Affine {
279                a: op.a * other.a % P,
280                b: (op.a * other.b + op.b) % P,
281            }
282        }
283    }
284
285    #[test]
286    fn test_affine() {
287        let mut rng = StdRng::seed_from_u64(42);
288        for _ in 0..200 {
289            let n = rng.gen_range(1..=40);
290            let q = rng.gen_range(1..=40);
291            let mut vec = (0..n)
292                .map(|_| Value {
293                    value: rng.gen_range(0..P),
294                    len: 1,
295                })
296                .collect::<Vec<_>>();
297            let mut seg = LazySegtree::<O>::new(&vec);
298            for _ in 0..q {
299                match rng.gen_range(0..2) {
300                    // range_apply
301                    0 => {
302                        let mut l = rng.gen_range(0..=n + 1);
303                        let mut r = rng.gen_range(0..=n);
304                        if l > r {
305                            (l, r) = (r, l - 1);
306                        }
307                        let f = Affine {
308                            a: rng.gen_range(0..P),
309                            b: rng.gen_range(0..P),
310                        };
311                        seg.range_apply(l..r, &f);
312                        for x in &mut vec[l..r] {
313                            *x = O::apply(&f, &*x);
314                        }
315                    }
316                    // fold
317                    1 => {
318                        let mut l = rng.gen_range(0..=n + 1);
319                        let mut r = rng.gen_range(0..=n);
320                        if l > r {
321                            (l, r) = (r, l - 1);
322                        }
323                        let result = seg.fold(l..r);
324                        let expected = vec[l..r]
325                            .iter()
326                            .fold(Value { value: 0, len: 0 }, |acc, &x| O::op(&acc, &x));
327                        assert_eq!(result, expected);
328                    }
329                    _ => unreachable!(),
330                }
331            }
332        }
333    }
