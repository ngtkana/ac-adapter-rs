12#[derive(Clone, Debug, PartialEq, serde::Serialize)]
13struct CrateMetadata {
14    dependencies: Vec<String>,
15    tags: Vec<String>,
16    description: Option<String>,

19#[derive(Clone, Debug, PartialEq, serde::Deserialize)]
20struct MetadataFile {
21    tags: Option<TagsSection>,
22    description: Option<DescriptionSection>,
23}

25#[derive(Clone, Debug, PartialEq, serde::Deserialize)]
26struct TagsSection {

30#[derive(Clone, Debug, PartialEq, serde::Deserialize)]
31struct DescriptionSection {

35// TODO: use crates.js etc. in target/doc/{crates.js,source-files.js} to bundle files
36// TODO: or use `.packages | map({ "name": .name, "dependencies": .dependencies })` of `cargo
37// metadata
38fn main() {
39    PROJECT_ROOT.set(find_project_root_path()).unwrap();
40    CRATE_METADATAS
41        .set(
42            PROJECT_ROOT
43                .get()
44                .unwrap()
45                .join(Path::new("libs"))
46                .read_dir()
47                .unwrap()
48                .filter_map(Result::ok)
49                .filter_map(|crate_entry| {
50                    let crate_path = crate_entry.path();
51                    let cargo_toml_path = crate_path.join("Cargo.toml");
52                    let metadata_path = crate_path.join("metadata.toml");
53
54                    if cargo_toml_path.exists() {
55                        let crate_name = crate_path
56                            .file_name()
57                            .unwrap()
58                            .to_string_lossy()
59                            .into_owned();
60                        let dependencies = parse_local_dependencies_from_cargo_toml(
61                            &fs::read_to_string(&cargo_toml_path).unwrap(),
62                        );
63
64                        // メタデータファイルからタグと説明を読み取る
65                        let (tags, description) = if metadata_path.exists() {
66                            let metadata_content =
67                                fs::read_to_string(&metadata_path).unwrap_or_default();
68                            let metadata: MetadataFile = toml::from_str(&metadata_content)
69                                .unwrap_or(MetadataFile {
70                                    tags: None,
71                                    description: None,
72                                });
73
74                            let tags = metadata.tags.map_or(Vec::new(), |t| t.list);
75                            let description = metadata.description.map(|d| d.short);
76
77                            (tags, description)
78                        } else {
79                            (Vec::new(), None)
80                        };
81
82                        Some((
83                            crate_name,
84                            CrateMetadata {
85                                dependencies,
86                                tags,
87                                description,
88                            },
89                        ))
90                    } else {
91                        None
92                    }
93                })
94                .collect(),
95        )
96        .unwrap();
97
98    let json = serde_json::to_string(CRATE_METADATAS.get().unwrap()).unwrap();
99    fs::create_dir_all(PROJECT_ROOT.get().unwrap().join("docs")).unwrap();
100    fs::write(
101        PROJECT_ROOT
102            .get()
103            .unwrap()
104            .join("docs")
105            .join("dependencies.js"),
106        format!("dependencies = {json}"),
107    )
108    .unwrap();
109}

111fn find_project_root_path() -> PathBuf {
112    std::env::current_dir()
113        .unwrap()
114        .ancestors()
115        .find(|&ancestor| {
116            ancestor.read_dir().unwrap().any(|entry| {
117                entry.unwrap().path().as_path().file_name() == Some(OsStr::new("Cargo.lock"))
118            })
119        })
120        .unwrap()
121        .to_path_buf()
122}

124fn parse_local_dependencies_from_cargo_toml(file_content: &str) -> Vec<String> {
125    match toml::from_str::<toml::Value>(file_content).unwrap() {
126        toml::Value::Table(table) => match table.get("dependencies") {
127            None => Vec::new(),
128            Some(toml::Value::Table(table)) => table
129                .iter()
130                .filter_map(|(key, value)| -> Option<String> {
131                    (match value {
132                        toml::Value::Table(table) => table.get("path").map(|value| match value {
133                            toml::Value::String(path) => path,
134                            _ => unreachable!(),
135                        }),
136                        toml::Value::String(path) => Some(path),
137                        _ => unreachable!(),
138                    })
139                    .and_then(|path| path.starts_with("../").then(|| key.clone()))
140                })
141                .collect_vec(),
142            _ => unreachable!(),
143        },
144        _ => unreachable!(),
145    }
