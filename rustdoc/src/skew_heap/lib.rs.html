1//! Meld のできるヒープ
2//!
3//! # 使い方
4//!
5//! ## ヒープ
6//!
7//! ```
8//! use skew_heap::meld;
9//! use skew_heap::SkewHeap;
10//!
11//! let mut heap = SkewHeap::new(); // new で構築
12//! heap.push(3); // push で挿入
13//! heap.push(4);
14//! heap.push(2);
15//!
16//! assert_eq!(heap.pop(), Some(4)); // pop で最大がでてきます。
17//!
18//! assert_eq!(heap.peek(), Some(&3)); // peek で次が見られます。
19//! assert_eq!(heap.pop(), Some(3));
20//! assert_eq!(heap.pop(), Some(2));
21//! assert_eq!(heap.pop(), None);
22//! ```
23//!
24//!
25//! ## Meld
26//!
27//! ```
28//! use skew_heap::meld;
29//! use skew_heap::SkewHeap;
30//!
31//! // フリー関数 `meld` を使う方法
32//! let heap_a = [0, 2, 4, 6].iter().copied().collect::<SkewHeap<_>>();
33//! let heap_b = [1, 3, 5, 7].iter().copied().collect::<SkewHeap<_>>();
34//! let mut heap = meld(heap_a, heap_b);
35//! assert_eq!(heap.into_sorted_vec(), vec![0, 1, 2, 3, 4, 5, 6, 7]);
36//!
37//! // メソッド `SkewHeap::meld` を使う方法
38//! let mut heap_a = [0, 2, 4, 6].iter().copied().collect::<SkewHeap<_>>();
39//! let mut heap_b = [1, 3, 5, 7].iter().copied().collect::<SkewHeap<_>>();
40//! heap_a.meld(heap_b);
41//! assert_eq!(heap_a.into_sorted_vec(), vec![0, 1, 2, 3, 4, 5, 6, 7]);
42//! ```
43//!
44//! # ありそうでないもの
45//!
46//! * `iter()`, `into_iter()`, `into_iter_sorted()`, `drain()`, `drain_sorted()`: 親ポインタないので面倒です。
47//! * `len(), is_empty()`: 長さ保持するの面倒です。
48//! * `peek_mut()`: 再挿入面倒です。

58/// Meld のできるヒープ
59#[derive(Clone, Hash, PartialEq)]
60pub struct SkewHeap<T>(Option<Box<SkeyHeapNode<T>>>);
61impl<T: Ord> Default for SkewHeap<T> {
62    fn default() -> Self {
63        Self::new()
64    }
65}
66impl<'a, A: 'a + Copy + Ord> Extend<&'a A> for SkewHeap<A> {
67    fn extend<T: IntoIterator<Item = &'a A>>(&mut self, iter: T) {
68        iter.into_iter().copied().for_each(|x| self.push(x));
69    }
70}
71impl<A: Ord> Extend<A> for SkewHeap<A> {
72    fn extend<T: IntoIterator<Item = A>>(&mut self, iter: T) {
73        iter.into_iter().for_each(|x| self.push(x));
74    }
75}
76impl<A: Ord> FromIterator<A> for SkewHeap<A> {
77    fn from_iter<T: IntoIterator<Item = A>>(iter: T) -> Self {
78        let mut heap = Self::new();
79        heap.extend(iter);
80        heap
81    }
82}
83impl<T: fmt::Debug + Ord> fmt::Debug for SkewHeap<T> {
84    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
85        let mut f = f.debug_list();
86        if let Some(heap) = self.0.as_ref() {
87            push_entries(heap, &mut f);
88        }
89        f.finish()
90    }
91}
92impl<T: Ord> SkewHeap<T> {
93    /// 新しく構築します。
94    pub fn new() -> Self {
95        Self(None)
96    }
97
98    /// 中身を殻にします。
99    pub fn clear(&mut self) {
100        *self = Self::new();
101    }
102
103    /// 要素一つからなる `SkewHeap` を構築します。
104    pub fn singleton(value: T) -> Self {
105        Self(Some(Box::new(SkeyHeapNode::singleton(value))))
106    }
107
108    /// 2 つの `SkewHeap` から、その合併を構築します。
109    ///
110    /// # 計算量
111    ///
112    /// O ( lg ( self.len(), rhs.len() ) )
113    ///
114    /// ただし `SkewHeap::len` メソッドはありません。（あの！？）
115    pub fn meld(&mut self, rhs: Self) {
116        *self = meld(take(self), rhs);
117    }
118
119    /// 要素を一つ、追加します。
120    pub fn push(&mut self, value: T) {
121        self.meld(Self::singleton(value));
122    }
123
124    /// 含んでいる最大の要素への参照を返します。
125    pub fn peek(&self) -> Option<&T> {
126        self.0.as_ref().map(|heap| &heap.value)
127    }
128
129    /// 含んでいる最大の要素を取り除き、それを返します。
130    pub fn pop(&mut self) -> Option<T> {
131        let me = take(self);
132        let SkeyHeapNode { left, right, value } = *me.0?;
133        *self = Self(meld_node(left, right));
134        Some(value)
135    }
136
137    /// ソート済みの `Vec` に変換します。
138    pub fn into_sorted_vec(mut self) -> Vec<T> {
139        let mut vec = Vec::new();
140        while let Some(x) = self.pop() {
141            vec.push(x);
142        }
143        vec.reverse();
144        vec
145    }
146}
147/// 2 つの `SkewHeap` から、その合併を構築します。
148///
149/// # 計算量
150///
151/// O ( lg ( self.len(), rhs.len() ) )
152///
153/// ただし `SkewHeap::len` メソッドはありません。（あの！？）
154pub fn meld<T: Ord>(a: SkewHeap<T>, b: SkewHeap<T>) -> SkewHeap<T> {
155    SkewHeap(meld_node(a.0, b.0))
156}

158#[derive(Clone, Default, Hash, PartialEq)]
159struct SkeyHeapNode<T> {
160    left: Option<Box<SkeyHeapNode<T>>>,
161    right: Option<Box<SkeyHeapNode<T>>>,
162    value: T,
163}
164impl<T: Ord + fmt::Debug> fmt::Debug for SkeyHeapNode<T> {
165    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
166        let mut f = f.debug_list();
167        push_entries(self, &mut f);
168        f.finish()
169    }
170}
171fn push_entries<T: fmt::Debug + Ord>(heap: &SkeyHeapNode<T>, f: &mut DebugList) {
172    f.entry(&heap.value);
173    if let Some(left) = heap.left.as_deref() {
174        push_entries(left, f);
175    }
176    if let Some(right) = heap.right.as_deref() {
177        push_entries(right, f);
178    }
179}
180impl<T: Ord> SkeyHeapNode<T> {
181    pub fn singleton(value: T) -> Self {
182        Self {
183            left: None,
184            right: None,
185            value,
186        }
187    }
188}
189fn meld_node<T: Ord>(
190    a: Option<Box<SkeyHeapNode<T>>>,
191    b: Option<Box<SkeyHeapNode<T>>>,
192) -> Option<Box<SkeyHeapNode<T>>> {
193    let [mut a, mut b] = match [a, b] {
194        [None, None] => return None,
195        [Some(a), None] => return Some(a),
196        [None, Some(b)] => return Some(b),
197        [Some(a), Some(b)] => [a, b],
198    };
199    if a.value < b.value {
200        swap(&mut a, &mut b);
201    }
202    a.right = meld_node(a.right, Some(b));
203    swap(&mut a.left, &mut a.right);
204    Some(a)
205}

207#[cfg(test)]
208mod tests {
209    use super::meld;
210    use super::SkewHeap;
211    use itertools::Itertools;
212    use rand::prelude::StdRng;
213    use rand::Rng;
214    use rand::SeedableRng;
215    use std::collections::BinaryHeap;

218    #[test]
219    fn test_heap() {
220        let mut rng = StdRng::seed_from_u64(42);
221        for _ in 0..20 {
222            let mut binary = BinaryHeap::new();
223            let mut skew = SkewHeap::new();
224            for _ in 0..1000 {
225                match rng.gen_range(0..3) {
226                    0 => {
227                        // push
228                        let x = rng.gen_range(0..10);
229                        binary.push(x);
230                        skew.push(x);
231                    }
232                    1 => {
233                        // pop
234                        let expected = binary.pop();
235                        let result = skew.pop();
236                        assert_eq!(expected, result);
237                    }
238                    2 => {
239                        // collect
240                        let mut expected = binary.iter().copied().collect_vec();
241                        let mut result = skew.clone().into_sorted_vec();
242                        expected.sort_unstable();
243                        result.sort_unstable();
244                        assert_eq!(expected, result);
245                    }
246                    _ => unreachable!(),
247                }
248            }
249        }
250    }
251
252    #[test]
253    fn test_meld() {
254        let mut rng = StdRng::seed_from_u64(42);
255        for _ in 0..20 {
256            let n = rng.gen_range(0..10);
257            let m = rng.gen_range(0..10);
258            let a = repeat_with(|| rng.gen_range(0..20)).take(n).collect_vec();
259            let b = repeat_with(|| rng.gen_range(0..20)).take(m).collect_vec();
260            let skew_a = a.iter().copied().collect::<SkewHeap<_>>();
261            let skew_b = b.iter().copied().collect::<SkewHeap<_>>();
262            let mut skew = meld(skew_a, skew_b);
263            let mut expected_to_be_sorted = Vec::new();
264            while let Some(x) = skew.pop() {
265                expected_to_be_sorted.push(x);
266            }
267            assert!(expected_to_be_sorted.windows(2).all(|v| v[0] >= v[1]));
268        }
269    }
