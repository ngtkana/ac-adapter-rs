19pub struct SplayTree<O: Op> {
20    root: *mut Node<O>,
21}
22impl<O: Op> SplayTree<O> {
23    pub fn new() -> Self {
24        Self {
25            root: ptr::null_mut(),
26        }
27    }
28
29    pub fn fold_all(&self) -> Option<&Value<O>> {
30        unsafe { Some(&self.root.as_ref()?.value) }
31    }
32}
33
34impl<O: RightMin> SplayTree<O>
35where
36    O::LeafValue: Ord,
37{
38    pub fn insert(&mut self, value: O::LeafValue) {
39        unsafe {
40            if self.root.is_null() {
41                self.root = alloc_from_leaf_value(value);
42            } else {
43                let (existing_leaf, dir) = lower_bound(self.root, &value);
44                self.root = insert_leaf_beside_a_leaf(existing_leaf, dir, value);
45            }
46        }
47    }
48
49    pub fn contains(&mut self, key: &O::LeafValue) -> bool {
50        unsafe {
51            if self.root.is_null() {
52                return false;
53            }
54            let (leaf, _) = lower_bound(self.root, key);
55            self.root = splay(leaf);
56            let Value::Leaf(ref leaf_value) = (*leaf).value else {
57                panic!("leaf is not a leaf");
58            };
59            leaf_value == key
60        }
61    }
62}
63
64impl<O: LeftLen> SplayTree<O> {
65    pub fn remove_at(&mut self, index: usize) -> O::LeafValue {
66        unsafe {
67            let leaf = get_by_index(self.root, index);
68            let (new_root, leaf_value) = remove_leaf(leaf);
69            self.root = new_root;
70            leaf_value
71        }
72    }
73}

81pub trait Op: Sized {
82    type InternalValue;
83    type LeafValue;
84
85    fn proj(x: &Self::LeafValue) -> Self::InternalValue;
86
87    fn mul(lhs: &Self::InternalValue, rhs: &Self::InternalValue) -> Self::InternalValue;
88}
89
90pub trait LeftLen: Op {
91    fn left_len(internal_value: &Self::InternalValue) -> usize;
92}
93
94pub trait RightMin: Op
95where
96    <Self as Op>::LeafValue: Ord,
97{
98    fn right_min(internal_value: &Self::InternalValue) -> &Self::LeafValue;
99}

101pub enum Value<O: Op> {
102    Leaf(O::LeafValue),
103    Internal(O::InternalValue),
104}
105impl<O: Op> std::fmt::Debug for Value<O>
106where
107    O::LeafValue: std::fmt::Debug,
108    O::InternalValue: std::fmt::Debug,
109{

118struct Node<O: Op> {
119    parent: *mut Self,
120    left: *mut Self,
121    right: *mut Self,
122    value: Value<O>,
123}
124impl<O: Op> Node<O> {
125    fn from_value(value: Value<O>) -> Self {
126        Self {
127            parent: ptr::null_mut(),
128            left: ptr::null_mut(),
129            right: ptr::null_mut(),
130            value,
131        }
132    }
133    unsafe fn update(&mut self) {
134        fn op_values<O: Op>(lhs: &Value<O>, rhs: &Value<O>) -> O::InternalValue {
135            match (lhs, rhs) {
136                (Value::Leaf(l), Value::Leaf(r)) => O::mul(&O::proj(l), &O::proj(r)),
137                (Value::Leaf(l), Value::Internal(r)) => O::mul(&O::proj(l), r),
138                (Value::Internal(l), Value::Leaf(r)) => O::mul(l, &O::proj(r)),
139                (Value::Internal(l), Value::Internal(r)) => O::mul(l, r),
140            }
141        }
142        self.value = Value::Internal(op_values(&(*self.left).value, &(*self.right).value));
143    }
144}

146fn alloc_from_node<O: Op>(node: Node<O>) -> *mut Node<O> {
147    Box::into_raw(Box::new(node))
148}
149fn alloc_from_value<O: Op>(value: Value<O>) -> *mut Node<O> {
150    alloc_from_node(Node::from_value(value))
151}
152fn alloc_from_leaf_value<O: Op>(leaf_value: O::LeafValue) -> *mut Node<O> {
153    alloc_from_value(Value::Leaf(leaf_value))
154}

156#[derive(Clone, Copy, Debug)]
157pub enum Direction {
158    Left,
159    Right,
160}
161
162unsafe fn insert_leaf_beside_a_leaf<O: Op>(
163    existing_leaf: *mut Node<O>, // not null
164    dir: Direction,
165    new_leaf_value: O::LeafValue,
166) -> *mut Node<O> {
167    let p = (*existing_leaf).parent;
168    let Value::Leaf(existing_leaf_value) = &(*existing_leaf).value else {
169        panic!("existing_leaf is not a leaf");
170    };
171    let new_leaf = alloc_from_leaf_value(new_leaf_value);
172    let Value::Leaf(new_leaf_value) = &(*new_leaf).value else { unreachable!() };
173    let internal = alloc_from_node(match dir {
174        Direction::Left => Node {
175            parent: p,
176            left: new_leaf,
177            right: existing_leaf,
178            value: Value::Internal(O::mul(
179                &O::proj(new_leaf_value),
180                &O::proj(existing_leaf_value),
181            )),
182        },
183        Direction::Right => Node {
184            parent: p,
185            left: existing_leaf,
186            right: new_leaf,
187            value: Value::Internal(O::mul(
188                &O::proj(existing_leaf_value),
189                &O::proj(new_leaf_value),
190            )),
191        },
192    });
193    (*existing_leaf).parent = internal;
194    (*new_leaf).parent = internal;
195    if !p.is_null() {
196        if (*p).left == existing_leaf {
197            (*p).left = internal;
198        } else {
199            (*p).right = internal;
200        }
201        (*p).update();
202    }
203    splay(internal)
204}

206unsafe fn remove_leaf<O: Op>(leaf: *mut Node<O>) -> (*mut Node<O>, O::LeafValue) {
207    let Node {
208        parent: p,
209        value: Value::Leaf(leaf_value),
210        ..
211    } = *Box::from_raw(leaf)
212    else {
213        panic!("leaf is not a leaf");
214    };
215    if p.is_null() {
216        return (ptr::null_mut(), leaf_value);
217    }
218    let s = if (*p).left == leaf { (*p).right } else { (*p).left };
219    let q = (*p).parent;
220    (*s).parent = q;
221    if q.is_null() {
222        (s, leaf_value)
223    } else {
224        if (*q).left == p {
225            (*q).left = s;
226        } else {
227            (*q).right = s;
228        }
229        (*q).update();
230        (splay(q), leaf_value)
231    }
232}

234unsafe fn lower_bound<O: RightMin>(
235    root: *mut Node<O>, // not null
236    key: &<O as Op>::LeafValue,
237) -> (*mut Node<O>, Direction)
238where
239    O::LeafValue: Ord,
240{
241    let mut x = root;
242    loop {
243        match (*x).value {
244            Value::Leaf(ref leaf_value) => {
245                let dir = if key <= leaf_value { Direction::Left } else { Direction::Right };
246                return (x, dir);
247            }
248            Value::Internal(ref internal_value) => {
249                x = if key <= O::right_min(internal_value) { (*x).left } else { (*x).right };
250            }
251        }
252    }
253}

255unsafe fn get_by_index<O: LeftLen>(
256    root: *mut Node<O>, // not null
257    mut index: usize,
258) -> *mut Node<O> {
259    let mut x = root;
260    loop {
261        match (*x).value {
262            Value::Leaf(_) => return x,
263            Value::Internal(ref internal_value) => {
264                let left_len = O::left_len(internal_value);
265                if index < left_len {
266                    x = (*x).left;
267                } else {
268                    index -= left_len;
269                    x = (*x).right;
270                }
271            }
272        }
273    }
274}

276unsafe fn splay<O: Op>(x: *mut Node<O>) -> *mut Node<O> {
277    let mut p = (*x).parent;
278    while !p.is_null() {
279        let g = (*p).parent;
280        if g.is_null() {
281            if (*p).left == x {
282                rotate_right(p);
283            } else {
284                rotate_left(p);
285            }
286        } else if (*g).left == p {
287            if (*p).left == x {
288                rotate_right(g);
289                rotate_right(p);
290            } else {
291                rotate_left(p);
292                rotate_right(g);
293            }
294        } else {
295            #[allow(clippy::collapsible_if)]
296            if (*p).left == x {
297                rotate_right(p);
298                rotate_left(g);
299            } else {
300                rotate_left(g);
301                rotate_left(p);
302            }
303        }
304        p = (*x).parent;
305    }
306    x
307}

309unsafe fn rotate_left<O: Op>(x: *mut Node<O>) {
310    let p = (*x).parent;
311    let y = (*x).right;
312    let c = (*y).left;
313    (*x).right = c;
314    if !c.is_null() {
315        (*c).parent = x;
316    }
317    (*y).left = x;
318    (*x).parent = y;
319    (*x).update();
320    (*y).update();
321    (*y).parent = p;
322    if !p.is_null() {
323        if (*p).left == x {
324            (*p).left = y;
325        } else {
326            (*p).right = y;
327        }
328        (*p).update();
329    }
330}
331
332unsafe fn rotate_right<O: Op>(x: *mut Node<O>) {
333    let p = (*x).parent;
334    let y = (*x).left;
335    let c = (*y).right;
336    (*x).left = c;
337    if !c.is_null() {
338        (*c).parent = x;
339    }
340    (*y).right = x;
341    (*x).parent = y;
342    (*x).update();
343    (*y).update();
344    (*y).parent = p;
345    if !p.is_null() {
346        if (*p).left == x {
347            (*p).left = y;
348        } else {
349            (*p).right = y;
350        }
351        (*p).update();
352    }
353}

368struct SplayTreeLeafFormatter<'a, O: Op> {
369    tree: &'a SplayTree<O>,
370}
371impl<O: Op> std::fmt::Debug for SplayTreeLeafFormatter<'_, O>
372where
373    O: Op,
374    O::LeafValue: std::fmt::Debug,
375{
376    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
377        unsafe fn find_leaves<O: Op>(
378            node: *mut Node<O>,
379            f: &mut std::fmt::DebugList,
380        ) -> std::fmt::Result
381        where
382            O::LeafValue: std::fmt::Debug,
383        {
384            if node.is_null() {
385                return Ok(());
386            }
387            find_leaves((*node).left, f)?;
388            if let Value::Leaf(ref value) = (*node).value {
389                f.entry(&value);
390            }
391            find_leaves((*node).right, f)
392        }
393        unsafe {
394            let mut f = f.debug_list();
395            find_leaves(self.tree.root, &mut f)?;
396            f.finish()?;
397            Ok(())
398        }
399    }
400}

402struct SplayTreeInternalFormatter<'a, O: Op> {
403    tree: &'a SplayTree<O>,
404}
405impl<O: Op> std::fmt::Debug for SplayTreeInternalFormatter<'_, O>
406where
407    O: Op,
408    O::InternalValue: std::fmt::Debug,
409{
410    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
411        unsafe fn find_internals<O: Op>(
412            node: *mut Node<O>,
413            found_leaves: &mut usize,
414            f: &mut std::fmt::DebugList,
415        ) -> std::fmt::Result
416        where
417            O::InternalValue: std::fmt::Debug,
418        {
419            if node.is_null() {
420                return Ok(());
421            }
422            match (*node).value {
423                Value::Leaf(_) => *found_leaves += 1,
424                Value::Internal(ref value) => {
425                    let start = *found_leaves;
426                    find_internals((*node).left, found_leaves, f)?;
427                    find_internals((*node).right, found_leaves, f)?;
428                    let end = *found_leaves;
429                    f.entry(&(start..end, value));
430                }
431            }
432            Ok(())
433        }
434        unsafe {
435            let mut f = f.debug_list();
436            find_internals(self.tree.root, &mut 0, &mut f)?;
437            f.finish()?;
438            Ok(())
439        }
440    }
