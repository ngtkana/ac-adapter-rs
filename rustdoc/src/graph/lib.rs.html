1//! Kosaraju's algorithm for strongly connected components
2//!
3//! # Note: CSR (Compressed Sparse Row)
4//!
5//! Graphs and set partitions are represented in CSR format [`Csr`].
6//!
7//!
8//! # Examples
9//! ```
10//! use graph::kosaraju;
11//! let n = 6;
12//! let edges = [ (1, 4), (5, 2), (3, 0), (5, 5), (4, 1), (0, 3), (4, 2)];
13//! let scc = kosaraju(n, &edges);
14//! assert_eq!(scc.len(), 4);
15//! assert_eq!(scc[0], [5]);
16//! assert_eq!(scc[1], [1, 4]);
17//! assert_eq!(scc[2], [2]);
18//! assert_eq!(scc[3], [0, 3]);
19//! ```
20mod csr;
21
22pub use csr::Csr;
23pub use csr::Iter;
24pub use csr::LastMut;

26/// Returns the strongly connected components of a directed graph.
27///
28/// # Examples
29/// ```
30/// use graph::kosaraju;
31/// let n = 6;
32/// let edges = [ (1, 4), (5, 2), (3, 0), (5, 5), (4, 1), (0, 3), (4, 2)];
33/// let scc = kosaraju(n, &edges);
34/// assert_eq!(scc.len(), 4);
35/// assert_eq!(scc[0], [5]);
36/// assert_eq!(scc[1], [1, 4]);
37/// assert_eq!(scc[2], [2]);
38/// assert_eq!(scc[3], [0, 3]);
39/// ```
40pub fn kosaraju(n: usize, edges: &[(usize, usize)]) -> Csr<usize> {
41    let (g, rg) = Csr::from_edges_and_rev(n, edges);
42    let mut sorted = Vec::with_capacity(n);
43    let mut cmp = vec![0; n];
44    for i in 0..n {
45        if cmp[i] == 0 {
46            dfs1(i, &g, &mut sorted, &mut cmp);
47        }
48    }
49    let mut out = Csr::with_capacity(n, n);
50    for &i in sorted.iter().rev() {
51        if cmp[i] == usize::MAX {
52            out.push_empty();
53            dfs2(i, &rg, &mut out, &mut cmp);
54        }
55    }
56    out
57}
58
59fn dfs1(i: usize, g: &Csr<usize>, sorted: &mut Vec<usize>, cmp: &mut [usize]) {
60    cmp[i] = usize::MAX;
61    for &y in &g[i] {
62        if cmp[y] == 0 {
63            dfs1(y, g, sorted, cmp);
64        }
65    }
66    sorted.push(i);
67}
68
69fn dfs2(i: usize, rg: &Csr<usize>, out: &mut Csr<usize>, cmp: &mut [usize]) {
70    cmp[i] = out.len();
71    out.last_mut().push(i);
72    for &y in &rg[i] {
73        if cmp[y] == usize::MAX {
74            dfs2(y, rg, out, cmp);
75        }
76    }
