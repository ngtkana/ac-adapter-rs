1//! 2-way tree DP
2//!
3//! # Graph-theoritic notation
4//!
5//! - $V$: vertex type
6//! - $T$: (rooted) tree type
7//! - $F$: (rooted) forest type (with ordering of the trees)
8//!
9//! # Algebraic notation
10//!
11//! $$
12//! \phi(T_i^\triangle) = g\left( \prod_{j \lessdot i} f(T_j^\triangle) \right)
13//! $$
14//!
15//! # Operations
16//!
17//! | | graph-theoritic | algebraic |
18//! | - | - | - |
19//! | `up` | $F \times V \to T \to F$: join by a root | $f \circ g$ |
20//! | `mul` | $F \times F \to F$: concatenate two forests | product of $M$ |
21//! | `identity` | $* \to F$: empty forest | identity of $M$ |
22//!
23//! # Return value
24//!
25//! | | description |
26//! | - | - |
27//! | `upper` | $f(\phi(T_i^\triangle))$ |
28//! | `lower` | $\prod_{j \lessdot i} f(\phi(T_j^\blacktriangledown))$ |
29//! | `branch` | $f(\phi(T_i^\blacktriangledown))$ |
30
31/// # Operations
32///
33/// | | graph-theoritic | algebraic |
34/// | - | - | - |
35/// | `up` | $F \times V \to T \to F$: join by a root | $f \circ g$ |
36/// | `mul` | $F \times F \to F$: concatenate two forests | product of $M$ |
37/// | `identity` | $* \to F$: empty forest | identity of $M$ |
38///
39/// # Return value
40///
41/// | | description |
42/// | - | - |
43/// | `upper` | $f(\phi(T_i^\triangle))$ |
44/// | `lower` | $\prod_{j \lessdot i} f(\phi(T_j^\blacktriangledown))$ |
45/// | `branch` | $f(\phi(T_i^\blacktriangledown))$ |
46pub trait Op: Sized {
47    /// A monoid $M$.
48    type Value: Clone;
49
50    /// $F \times V \to T \to F$: join by a root
51    fn up(&self, value: &Self::Value, root: usize) -> Self::Value;
52
53    /// $F \times F \to F$: concatenate two forests
54    fn mul(&self, lhs: &Self::Value, rhs: &Self::Value) -> Self::Value;
55
56    /// $* \to F$: empty forest
57    fn identity(&self) -> Self::Value;
58
59    /// Performs 2-way tree DP
60    fn two_way_tree_fold(
61        &self,
62        g: &[Vec<usize>],
63        sorted: &[usize],
64    ) -> TwoWayTreeFoldResult<Self::Value> {
65        two_way_tree_fold(self, g, sorted)
66    }
67}

69/// The return value of [`Op::two_way_tree_fold()`] ans [`two_way_tree_fold()`]
70pub struct TwoWayTreeFoldResult<T> {
71    /// $f(\phi(T_i^\triangle))$
72    pub upper: Vec<T>,
73    /// $\prod_{j \lessdot i} f(\phi(T_j^\blacktriangledown))$
74    pub lower: Vec<T>,
75    /// $f(\phi(T_i^\blacktriangledown))$
76    pub branch: Vec<T>,
77}

79/// Performs 2-way tree DP
80///
81/// # Operations
82///
83/// | | graph-theoritic | algebraic |
84/// | - | - | - |
85/// | `up` | $F \times V \to T \to F$: join by a root | $f \circ g$ |
86/// | `mul` | $F \times F \to F$: concatenate two forests | product of $M$ |
87/// | `identity` | $* \to F$: empty forest | identity of $M$ |
88///
89/// # Return value
90///
91/// | | description |
92/// | - | - |
93/// | `upper` | $f(\phi(T_i^\triangle))$ |
94/// | `lower` | $\prod_{j \lessdot i} f(\phi(T_j^\blacktriangledown))$ |
95/// | `branch` | $f(\phi(T_i^\blacktriangledown))$ |
96pub fn two_way_tree_fold<O: Op>(
97    o: &O,
98    g: &[Vec<usize>],
99    sorted: &[usize],
100) -> TwoWayTreeFoldResult<O::Value> {
101    let n = g.len();
102    let mut lower = vec![o.identity(); n];
103    let mut branch = vec![o.identity(); n];
104    for &i in sorted.iter().rev() {
105        for &j in &g[i] {
106            lower[i] = o.mul(&lower[i], &branch[j]);
107        }
108        branch[i] = o.up(&lower[i], i);
109    }
110    let mut upper = vec![o.identity(); n];
111    for &i in sorted {
112        let mut suffix = upper[i].clone();
113        for &j in g[i].iter().rev() {
114            upper[j] = suffix.clone();
115            suffix = o.mul(&branch[j], &suffix);
116        }
117        let mut prefix = o.identity();
118        for &j in &g[i] {
119            upper[j] = o.up(&o.mul(&upper[j], &prefix), i);
120            prefix = o.mul(&prefix, &branch[j]);
121        }
122    }
123    TwoWayTreeFoldResult {
124        upper,
125        lower,
126        branch,
127    }
128}

132    use super::*;
133    use rand::rngs::StdRng;
134    use rand::Rng;
135    use rand::SeedableRng;
136    use std::collections::BinaryHeap;
137
138    struct GenratedTree {
139        g: Vec<Vec<usize>>,
140        sorted: Vec<usize>,
141        parent: Vec<usize>,
142    }
143
144    fn gen_tree(rng: &mut StdRng, n: usize) -> GenratedTree {
145        let mut g = vec![Vec::new(); n];
146        let rev_prufer = (0..n - 1)
147            .map(|i| if i == n - 2 { 0 } else { rng.gen_range(0..n) })
148            .collect::<Vec<_>>();
149        let mut count = vec![0; n];
150        for &y in &rev_prufer {
151            count[y] += 1;
152        }
153        let mut heap = (0..n).filter(|&y| count[y] == 0).collect::<BinaryHeap<_>>();
154        for &y in &rev_prufer {
155            let x = heap.pop().unwrap();
156            g[x].push(y);
157            g[y].push(x);
158            count[y] -= 1;
159            if count[y] == 0 {
160                heap.push(y);
161            }
162        }
163        let root = rng.gen_range(0..n);
164        let mut stack = vec![root];
165        let mut parent = vec![usize::MAX; n];
166        let mut sorted = Vec::new();
167        parent[root] = root;
168        while let Some(i) = stack.pop() {
169            sorted.push(i);
170            g[i].retain(|&j| parent[i] != j);
171            for &j in &g[i] {
172                stack.push(j);
173                parent[j] = i;
174            }
175        }
176        GenratedTree { g, sorted, parent }
177    }

179    #[test]
180    fn test_subtree() {
181        struct O;
182        #[derive(Clone, PartialEq)]
183        struct Tree {
184            root: usize,
185            child: Vec<Self>,
186        }
187        impl std::fmt::Debug for Tree {
188            fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
189                f.debug_tuple("")
190                    .field(&self.root)
191                    .field(&self.child)
192                    .finish()
193            }
194        }
195        impl Op for O {
196            type Value = Vec<Tree>;
197
198            fn up(&self, value: &Self::Value, root: usize) -> Self::Value {
199                vec![Tree {
200                    root,
201                    child: value.clone(),
202                }]
203            }
204
205            fn identity(&self) -> Self::Value {
206                Vec::new()
207            }
208
209            fn mul(&self, lhs: &Self::Value, rhs: &Self::Value) -> Self::Value {
210                lhs.iter().cloned().chain(rhs.iter().cloned()).collect()
211            }
212        }
213
214        fn subtree(i: usize, p: usize, g: &[Vec<usize>]) -> Tree {
215            Tree {
216                root: i,
217                child: match g[i].iter().position(|&j| j == p) {
218                    None => g[i].iter().map(|&j| subtree(j, i, g)).collect(),
219                    Some(e) => g[i][e + 1..]
220                        .iter()
221                        .chain(&g[i][..e])
222                        .map(|&j| subtree(j, i, g))
223                        .collect(),
224                },
225            }
226        }
227
228        let mut rng = StdRng::seed_from_u64(42);
229        for _ in 0..100 {
230            let n = rng.gen_range(1..=30);
231            let GenratedTree {
232                g, sorted, parent, ..
233            } = gen_tree(&mut rng, n);
234            let TwoWayTreeFoldResult {
235                lower,
236                branch,
237                upper,
238            } = O.two_way_tree_fold(&g, &sorted);
239            let mut g1 = g.clone();
240            for (i, &p) in parent.iter().enumerate() {
241                if i != p {
242                    g1[i].push(p);
243                }
244            }
245            for (i, &p) in parent.iter().enumerate() {
246                assert_eq!(
247                    lower[i],
248                    g[i].iter().map(|&j| subtree(j, i, &g)).collect::<Vec<_>>()
249                );
250                assert_eq!(branch[i], vec![subtree(i, p, &g)]);
251                assert_eq!(
252                    upper[i],
253                    if i == p { Vec::new() } else { vec![subtree(p, i, &g1)] }
254                );
255            }
256        }
257    }
