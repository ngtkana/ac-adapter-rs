3/// Returns the next permutation of `a` in lexicographic order.
4pub fn next_permutation<T: Ord>(a: &mut [T]) -> bool {
5    let Some(i) = a.windows(2).rposition(|w| w[0] < w[1]) else {
6        return false;
7    };
8    let j = a.iter().rposition(|x| x > &a[i]).unwrap();
9    a.swap(i, j);
10    a[i + 1..].reverse();
11    true
12}
13/// Calls `f` for each permutation of `a` in lexicographic order.
14pub fn for_each_permutation<T: Ord, F: FnMut(&[T])>(a: &mut [T], mut f: F) {
15    a.sort();
16    while {
17        f(a);
18        next_permutation(a)
19    } {}
20}
21/// Returns all permutations of `a` in lexicographic order.
22pub fn permutations<T: Ord + Clone>(a: Vec<T>) -> Vec<Vec<T>> {
23    let mut a = a;
24    let mut result = Vec::new();
25    for_each_permutation(&mut a, |a| result.push(a.to_vec()));
26    result
27}
28/// Returns the next $(K, N - K)$-shuffle of `a` in lexicographic order.
29pub fn next_shuffle<T: Ord>(a: &mut [T], k: usize) -> bool {
30    let n = a.len();
31    if n == k {
32        return false;
33    }
34    let (left, right) = a.split_at_mut(k);
35    let Some(mut i) = left.iter().rposition(|x| x < right.last().unwrap()) else {
36        return false;
37    };
38    let mut j = right.iter().position(|x| &left[i] < x).unwrap();
39    std::mem::swap(&mut left[i], &mut right[j]);
40    i += 1;
41    j += 1;
42    let swap_len = (k - i).min(n - k - j);
43    left[k - swap_len..].swap_with_slice(&mut right[j..j + swap_len]);
44    left[i..].rotate_left(k.saturating_sub(i + swap_len));
45    right[j..].rotate_right((n - k).saturating_sub(j + swap_len));
46    true
47}
48/// Calls `f` for each $(K, N - K)$-shuffle of `a` in lexicographic order.
49pub fn for_each_shuffle<T: Ord, F: FnMut(&[T])>(a: &mut [T], k: usize, mut f: F) {
50    a.sort();
51    while {
52        f(&a[..k]);
53        next_shuffle(a, k)
54    } {}
55}
56/// Returns all $(K, N - K)$-shuffles of `a` in lexicographic order.
57pub fn shuffles<T: Ord + Clone>(a: Vec<T>, k: usize) -> Vec<Vec<T>> {
58    let mut a = a;
59    let mut result = Vec::new();
60    for_each_shuffle(&mut a, k, |a| result.push(a.to_vec()));
61    result
62}
63/// Returns the next pairing (ascending $(2,2,\dots,2)$-shuffle) of `a` in lexicographic order.
64pub fn next_pairing(a: &mut [usize]) -> bool {
65    assert!(a.len() % 2 == 0);
66    assert!(a.len() <= 32);
67    let n = a.len();
68    let mut used = 0_u32;
69    for i in (0..n).rev() {
70        used |= 1 << a[i];
71        if i % 2 == 1 && a[i] + 1 < (used + 1).next_power_of_two().trailing_zeros() as usize {
72            a[i] = (used >> (a[i] + 1)).trailing_zeros() as usize + a[i] + 1;
73            used ^= 1 << a[i];
74            for x in &mut a[i + 1..n] {
75                *x = used.trailing_zeros() as usize;
76                used ^= 1 << *x;
77            }
78            return true;
79        }
80    }
81    false
82}
83/// Calls `f` for each pairing (ascending $(2,2,\dots,2)$-shuffle) of `a` in lexicographic order.
84pub fn for_each_pairing<F: FnMut(&[usize])>(n: usize, mut f: F) {
85    let mut a = (0..n).collect::<Vec<_>>();
86    while {
87        f(&a);
88        next_pairing(&mut a)
89    } {}
90}
91/// Returns all pairings (ascending $(2,2,\dots,2)$-shuffles) of `a` in lexicographic order.
92pub fn pairings(n: usize) -> Vec<Vec<usize>> {
93    let mut result = Vec::new();
94    for_each_pairing(n, |a| result.push(a.to_vec()));
95    result
96}
97/// Returns the next partition of `a` in lexicographic order.
98pub fn next_partition(a: &mut Vec<usize>) -> bool {
99    let Some(mut sum) = a.pop() else { return false };
100    if a.is_empty() {
101        return false;
102    }
103    while let Some(x) = a.pop() {
104        sum += x;
105        if a.last().is_none_or(|&last| last > x) {
106            a.push(x + 1);
107            a.extend(std::iter::repeat_n(1, sum - x - 1));
108            break;
109        }
110    }
111    true
112}
113/// Calls `f` for each partition of `n` in lexicographic order.
114pub fn for_each_partition<F: FnMut(&[usize])>(n: usize, mut f: F) {
115    let mut a = vec![1; n];
116    while {
117        f(&a);
118        next_partition(&mut a)
119    } {}
120}
121/// Returns all partitions of `n` in lexicographic order.
122pub fn partitions(n: usize) -> Vec<Vec<usize>> {
123    let mut result = Vec::new();
124    for_each_partition(n, |a| result.push(a.to_vec()));
125    result
126}
127/// Returns the previous partition of `a` in lexicographic order.
128pub fn prev_partition(a: &mut Vec<usize>) -> bool {
129    let Some(i) = a.iter().rposition(|&x| x != 1) else { return false };
130    let max = a[i] - 1;
131    let mut sum = a.split_off(i).into_iter().sum::<usize>();
132    while sum >= max {
133        a.push(max);
134        sum -= max;
135    }
136    if sum > 0 {
137        a.push(sum);
138    }
139    true
140}
141/// Calls `f` for each partition of `n` in the reverse of the lexicographic order.
142pub fn for_each_partition_rev<F: FnMut(&[usize])>(n: usize, mut f: F) {
143    if n == 0 {
144        f(&[]);
145        return;
146    }
147    let mut a = vec![n];
148    while {
149        f(&a);
150        prev_partition(&mut a)
151    } {}
152}
153/// Returns all partitions of `n` in the reverse of the lexicographic order.
154pub fn partitions_rev(n: usize) -> Vec<Vec<usize>> {
155    let mut result = Vec::new();
156    for_each_partition_rev(n, |a| result.push(a.to_vec()));
157    result
158}

160#[cfg(test)]
161mod tests {
162    use super::*;
163    use itertools::Itertools;
164
165    #[test]
166    fn test_permutations() {
167        for n in 0..=5 {
168            let result = permutations((0..n).collect::<Vec<_>>());
169            assert!(result.iter().tuple_windows().all(|(a, b)| a < b));
170            assert_eq!(result.len(), (1..=n).product());
171        }
172    }
173
174    #[test]
175    fn test_shuffles() {
176        for n in 0..=5 {
177            for k in 0..=n {
178                let result = shuffles((0..n).collect::<Vec<_>>(), k);
179                for result in &result {
180                    assert!(result[..k].iter().tuple_windows().all(|(a, b)| a < b));
181                    assert!(result[k..].iter().tuple_windows().all(|(a, b)| a < b));
182                }
183                assert!(result.iter().tuple_windows().all(|(a, b)| a < b));
184                assert_eq!(
185                    result.len(),
186                    (1..=n).product::<usize>()
187                        / (1..=k).product::<usize>()
188                        / (1..=(n - k)).product::<usize>()
189                );
190            }
191        }
192    }
193
194    #[test]
195    fn test_pairings() {
196        for n in (0..=10).step_by(2) {
197            let result = pairings(n);
198            assert!(result.iter().tuple_windows().all(|(a, b)| a < b));
199            for result in &result {
200                assert!(result.chunks_exact(2).all(|x| x[0] < x[1]));
201                assert!(result.iter().step_by(2).tuple_windows().all(|(a, b)| a < b));
202            }
203            assert_eq!(result.len(), (1..=n).step_by(2).product::<usize>());
204        }
205    }
206
207    #[test]
208    fn test_partitions() {
209        let n = 10;
210        let mut p = vec![1; n];
211        for step in 2..n {
212            for i in 0..n - step {
213                p[i + step] += p[i];
214            }
215        }
216        for (n, &p) in p.iter().enumerate() {
217            let result = partitions(n);
218            assert!(result.iter().tuple_windows().all(|(a, b)| a < b));
219            assert_eq!(result.len(), p);
220            assert_eq!(result, partitions_rev(n).into_iter().rev().collect_vec());
221        }
222    }
