1//! Dijkstra 法の最短路反復により、最小費用流問題を解きます。
2//!
3//! [See the document of `MinCostFlow`](MinCostFlow)
4
5use std::cmp::Reverse;
6use std::collections::BinaryHeap;
7use std::fmt::Debug;
8use std::mem::replace;
9
10/// [`MinCostFlow::get_edge`] の戻り値型
11#[derive(Clone, Default, Hash, PartialEq, Eq, Copy)]
12pub struct Edge {
13    from: usize,
14    to: usize,
15    flow: i64,
16    cap: i64,
17    cost: i64,
18}
19impl Debug for Edge {
20    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
21        fn fmt_i64(x: i64) -> String {
22            if x == i64::MAX {
23                "_".to_owned()
24            } else {
25                x.to_string()
26            }
27        }
28        let &Self {
29            from,
30            to,
31            flow,
32            cap,
33            cost,
34        } = self;
35        write!(
36            f,
37            "({}->{}, {}/{}, {})",
38            from,
39            to,
40            fmt_i64(flow),
41            fmt_i64(cap),
42            fmt_i64(cost),
43        )
44    }
45}
46#[derive(Clone, Debug, Default, Hash, PartialEq, Copy)]
47struct __InternalEdge {
48    to: usize,
49    rev: usize,
50    cap: i64,
51    cost: i64,
52}
53
54/// Dijkstra 法の最短路反復により、最小費用流問題を解きます。
55///
56/// # 使い方
57///
58/// - [`new()`](MinCostFlow::new) で空グラフを構築してき、
59/// - [`add_edge()`](MinCostFlow::add_edge) で辺を挿入していき、
60/// - [`flow()`](MinCostFlow::flow) または [`slope()`](MinCostFlow::slope) で答えを聞きます。
61///
62/// ```
63/// # use mincost_flow::MinCostFlow;
64/// let mut mcf = MinCostFlow::new(4);
65/// mcf.add_edge(0, 1, 1, 20); // from, to, cap, cost
66/// mcf.add_edge(0, 2, 1, 20);
67/// mcf.add_edge(1, 2, 1, 10);
68///
69/// let slope = mcf.slope(0, 2, i64::MAX);
70/// assert_eq!(slope, vec![(0, 0), (1, 20), (2, 50)]);
71/// ```
72///
73///
74/// また [`Debug`] トレイトを独自実装しています。
75#[derive(Clone, Default, Hash, PartialEq)]
76pub struct MinCostFlow {
77    g: Vec<Vec<__InternalEdge>>,
78    edge_position: Vec<[usize; 2]>,
79}
80impl Debug for MinCostFlow {
81    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
82        f.debug_list()
83            .entries((0..self.g.len()).map(|i| self.get_edge(i)))
84            .finish()
85    }
86}
87
88impl MinCostFlow {
89    /// 空グラフを構築します。
90    pub fn new(n: usize) -> Self {
91        Self {
92            g: vec![Vec::new(); n],
93            edge_position: Vec::new(),
94        }
95    }
96
97    /// 辺を追加します。
98    pub fn add_edge(&mut self, u: usize, v: usize, cap: i64, cost: i64) -> usize {
99        let res = self.edge_position.len();
100        let su = self.g[u].len();
101        let sv = self.g[v].len();
102        self.g[u].push(__InternalEdge {
103            to: v,
104            rev: sv,
105            cap,
106            cost,
107        });
108        self.g[v].push(__InternalEdge {
109            to: u,
110            rev: su,
111            cap: 0,
112            cost: -cost,
113        });
114        self.edge_position.push([u, su]);
115        res
116    }
117
118    /// `i` 番目に挿入した辺を取得します。
119    pub fn get_edge(&self, i: usize) -> Edge {
120        assert!(i < self.edge_position.len());
121        let [from, i] = self.edge_position[i];
122        let e = self.g[from][i];
123        let to = e.to;
124        let rev = &self.g[to][e.rev];
125        Edge {
126            from,
127            to,
128            cap: e.cap + rev.cap,
129            flow: rev.cap,
130            cost: e.cost,
131        }
132    }
133
134    /// 解きます
135    ///
136    /// # 出力形式
137    ///
138    /// `(flow, cost)`
139    pub fn flow(&mut self, source: usize, sink: usize, flow_limit: i64) -> (i64, i64) {
140        self.slope(source, sink, flow_limit).pop().unwrap()
141    }
142
143    /// 解きます
144    ///
145    /// # 出力形式
146    ///
147    /// `(flow, cost)`
148    pub fn slope(&mut self, source: usize, sink: usize, flow_limit: i64) -> Vec<(i64, i64)> {
149        let n = self.g.len();
150        let mut slope = vec![(0, 0)];
151        let mut flow = 0;
152        let mut cost = 0;
153        let mut prev_price = i64::MAX;
154        while flow < flow_limit {
155            let mut dual = vec![0; n];
156            let mut used = vec![false; n];
157            let mut prev = vec![!0; n];
158            let mut pree = vec![!0; n];
159            if !self.refine_dual_dijkstra(source, sink, &mut dual, &mut used, &mut prev, &mut pree)
160            {
161                break;
162            }
163
164            // 増分流量 aug の計算
165            let mut aug = flow_limit - flow;
166            let mut crr = sink;
167            while crr != source {
168                aug = aug.min(self.g[prev[crr]][pree[crr]].cap);
169                crr = prev[crr];
170            }
171
172            // ネットワークの更新
173            let mut crr = sink;
174            while crr != source {
175                self.g[prev[crr]][pree[crr]].cap -= aug;
176                let rev = self.g[prev[crr]][pree[crr]].rev;
177                self.g[crr][rev].cap += aug;
178                crr = prev[crr];
179            }
180
181            // Min-cost slope への書き込み
182            let price = -dual[source];
183            flow += aug;
184            cost += price * aug;
185            if replace(&mut prev_price, price) == price {
186                slope.pop().unwrap();
187            }
188            slope.push((flow, cost));
189        }
190        slope
191    }
192
193    fn refine_dual_dijkstra(
194        &self,
195        source: usize,
196        sink: usize,
197        dual: &mut [i64],
198        used: &mut [bool],  // キューから出したことがあれば、`true`
199        prev: &mut [usize], // 前者頂点を指す復元配列
200        pree: &mut [usize], // 前者辺の前者頂点からの隣接リストに於ける位置を指す復元配列
201    ) -> bool {
202        let mut heap = BinaryHeap::<(Reverse<i64>, usize)>::new();
203        heap.push((Reverse(0), source));
204        let mut dist = vec![i64::MAX; self.g.len()];
205        dist[source] = 0;
206        while let Some((Reverse(dx), x)) = heap.pop() {
207            used[x] = true;
208            if x == sink {
209                break;
210            }
211            for (i, &e) in self.g[x].iter().enumerate() {
212                let y = e.to;
213                if e.cap == 0 {
214                    continue;
215                }
216                let dy = dx.saturating_add(e.cost) + dual[x] - dual[y];
217                if dy < dist[y] {
218                    prev[y] = x;
219                    pree[y] = i;
220                    dist[y] = dy;
221                    heap.push((Reverse(dy), y));
222                }
223            }
224        }
225        if !used[sink] {
226            return false;
227        }
228        for i in (0..self.g.len()).filter(|&i| used[i]) {
229            dual[i] -= dist[sink] - dist[i];
230        }
231        true
232    }
233}

235#[cfg(test)]
236mod tests {
237    use super::MinCostFlow;
238
239    #[test]
240    fn test_mincost_flow() {
241        let mut graph = MinCostFlow::new(4);
242        assert_eq!(graph.add_edge(0, 1, 2, 1), 0);
243        assert_eq!(graph.add_edge(0, 2, 1, 2), 1);
244        assert_eq!(graph.add_edge(1, 2, 1, 1), 2);
245        assert_eq!(graph.add_edge(1, 3, 1, 3), 3);
246        assert_eq!(graph.add_edge(2, 3, 2, 1), 4);
247        assert_eq!(graph.flow(0, 3, 2), (2, 6));
248    }
249
250    #[test]
251    fn test_reduced_slope() {
252        let mut graph = MinCostFlow::new(3);
253        assert_eq!(graph.add_edge(0, 1, 1, 1), 0);
254        assert_eq!(graph.add_edge(1, 2, 1, 0), 1);
255        assert_eq!(graph.add_edge(0, 2, 2, 1), 2);
256        assert_eq!(&graph.slope(0, 2, i64::MAX), &[(0, 0), (3, 3)]);
257    }
258
259    #[test]
260    fn test_only_one_nonzero_cost_edge() {
261        let mut graph = MinCostFlow::new(3);
262        assert_eq!(graph.add_edge(0, 1, 1, 1), 0);
263        assert_eq!(graph.add_edge(1, 2, 1, 0), 1);
264        assert_eq!(graph.slope(0, 2, i64::MAX), &[(0, 0), (1, 1)]);
265    }
