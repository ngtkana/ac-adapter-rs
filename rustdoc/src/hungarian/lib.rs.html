1//! Solve an assignment problem by Hungarian algorithm.
2//!
3//! # Example
4//!
5//! Basic usage:
6//!
7//! ```
8//! use hungarian::hungarian;
9//!
10//! let result = hungarian(&[vec![2, 100, 10], vec![10, 100, 15]]);
11//!
12//! assert_eq!(result.value, 17);
13//! assert_eq!(&*result.forward, vec![0, 2].as_slice());
14//! assert_eq!(&*result.backward, vec![Some(0), None, Some(1)].as_slice());
15//! ```

23/// [See the crate level documentation](crate)
24pub fn hungarian<T: Value>(cost_matrix: &[Vec<T>]) -> HungarianResult<T> {
25    let h = cost_matrix.len();
26    let w = cost_matrix[0].len();
27    let mut m = vec![usize::MAX; w + 1];
28
29    // initialize a feasible potential
30    let mut all_min = T::infinity();
31    let mut left = vec![T::infinity(); h].into_boxed_slice();
32    for (i, x) in cost_matrix
33        .iter()
34        .enumerate()
35        .flat_map(|(i, v)| v.iter().map(move |&x| (i, x)))
36    {
37        if x < all_min {
38            all_min = x;
39        }
40        if x < left[i] {
41            left[i] = x;
42        }
43    }
44    left.iter_mut().for_each(|x| *x -= all_min);
45    let mut right = vec![T::zero(); w].into_boxed_slice();
46
47    for s in 0..h {
48        // dijkstra
49        let (pi, mut y) = {
50            let mut used_l = vec![false; h];
51            let mut used_r = vec![false; w + 1];
52            let mut pi = vec![w; w];
53            let mut dist = vec![T::infinity(); w];
54            m[w] = s;
55            let mut x0 = s;
56            let mut y0 = w;
57            while x0 != usize::MAX {
58                // find y0
59                let (swap, delta) = {
60                    let mut swap = w;
61                    used_r[y0] = true;
62                    used_l[x0] = true;
63                    let mut delta = T::infinity();
64                    for y in (0..w).filter(|&y| !used_r[y]) {
65                        let cost = cost_matrix[x0][y] + left[x0] - right[y];
66                        if cost < dist[y] {
67                            pi[y] = y0;
68                            dist[y] = cost;
69                        }
70                        if dist[y] < delta {
71                            swap = y;
72                            delta = dist[y];
73                        }
74                    }
75                    (swap, delta)
76                };
77
78                // update x0, y0
79                y0 = swap;
80                x0 = m[y0];
81
82                // update potential
83                for i in (0..h).filter(|&i| !used_l[i]) {
84                    left[i] += delta;
85                }
86                for j in (0..w).filter(|&i| !used_r[i]) {
87                    right[j] += delta;
88                    dist[j] -= delta;
89                }
90            }
91            (pi, y0)
92        };
93        // augmentation
94        while y != w {
95            m[y] = m[pi[y]];
96            y = pi[y];
97        }
98    }
99    m.pop();
100
101    let backward = m;
102    let mut forward = vec![usize::MAX; h].into_boxed_slice();
103    let mut value = T::zero();
104    for (y, &x) in backward.iter().enumerate() {
105        if x != usize::MAX {
106            forward[x] = y;
107            value += cost_matrix[x][y];
108        }
109    }
110    let backward = backward
111        .into_iter()
112        .map(|x| if x == usize::MAX { None } else { Some(x) })
113        .collect::<Vec<_>>()
114        .into_boxed_slice();
115    HungarianResult {
116        forward,
117        backward,
118        left,
119        right,
120        value,
121    }
122}

124/// A value object to represent the optimal solution of an assignment problem.
125#[derive(Debug, Clone, PartialEq, Eq)]
126pub struct HungarianResult<T: Value> {
127    /// Takes the first component of a match and returns the second one.
128    pub forward: Box<[usize]>,
129    /// Takes the second component of a match and returns the first one.
130    pub backward: Box<[Option<usize>]>,
131    /// A left half of an optimal potential.
132    pub left: Box<[T]>,
133    /// A right half of an optimal potential.
134    pub right: Box<[T]>,
135    /// The value of an optimal solution.
136    pub value: T,
137}
138
139/// A trait to adapt a value type to [`hungarian`]
140///
141/// This trait is already implemented for all the signed and unsigned integer types.
142pub trait Value:
143    Sized + Copy + Add<Output = Self> + AddAssign + Sub<Output = Self> + SubAssign + Sum + PartialOrd
144{
145    fn zero() -> Self;
146    fn infinity() -> Self;
147}
148
149macro_rules! impl_value_int {
150    ($($T:ident),* $(,)?) => {$(
151        impl Value for $T {
152            fn zero() -> Self {
153                0
154            }
155            fn infinity() -> Self {
156                $T::MAX
157            }
158        }
159    )*}
160}
161impl_value_int! {
162    u8, u16, u32, u64, u128, usize,
163    i8, i16, i32, i64, i128, isize,
164}

166macro_rules! impl_value_float {
167    ($($T:ident),* $(,)?) => {$(
168        impl Value for $T {
169            fn zero() -> Self {
170                0.
171            }
172            fn infinity() -> Self {
173                $T::INFINITY
174            }
175        }
176    )*}
177}
178impl_value_float!(f32, f64);

225    fn test_rand_impl<T: Value + Debug + Epsilon + SampleUniform>(
226        h: usize,
227        w: usize,
228        iter: usize,
229        brute: bool,
230        value_range: RangeInclusive<T>,
231    ) {
232        let mut rng = StdRng::seed_from_u64(42);
233        for _ in 0..iter {
234            let mut h = rng.gen_range(1..=h);
235            let mut w = rng.gen_range(1..=w);
236            if h > w {
237                swap(&mut h, &mut w);
238            }
239            let cost_matrix = repeat_with(|| {
240                repeat_with(|| rng.gen_range(value_range.clone()))
241                    .take(w)
242                    .collect_vec()
243            })
244            .take(h)
245            .collect_vec();
246            let result = hungarian(&cost_matrix);
247            if brute {
248                compare_with_brute(&cost_matrix, &result);
249            }
250            validate(&cost_matrix, &result);
251        }
252    }

254    fn validate<T: Value + Debug + Epsilon>(cost_matrix: &[Vec<T>], result: &HungarianResult<T>) {
255        let h = cost_matrix.len();
256        let w = cost_matrix[0].len();
257        let HungarianResult {
258            left,
259            right,
260            forward,
261            backward,
262            value,
263        } = result;
264
265        // partial 1 on 1 correspondence
266        assert_eq!(backward.iter().filter(|&x| x.is_some()).count(), h);
267        for i in 0..h {
268            assert_eq!(i, backward[forward[i]].unwrap());
269        }
270        for j in 0..w {
271            if let Some(i) = backward[j] {
272                assert_eq!(j, forward[i]);
273            }
274        }
275
276        // feasibility
277        for (i, j, x) in cost_matrix
278            .iter()
279            .enumerate()
280            .flat_map(|(i, v)| v.iter().enumerate().map(move |(j, &x)| (i, j, x)))
281        {
282            assert!(
283                right[j] <= x + left[i] + T::epsilon(),
284                "left[{i}] = {left:?}, right[{j}] = {right:?}, cost_matrix[{i}][{j}] = {cost:?}",
285                i = i,
286                left = left[i],
287                j = j,
288                right = right[j],
289                cost = x,
290            );
291        }
292
293        // saturation
294        for (i, &j) in forward.iter().enumerate() {
295            assert_abs_diff_eq!(
296                cost_matrix[i][j],
297                right[j] - left[i],
298                epsilon = T::epsilon()
299            );
300        }
301
302        // value is correct
303        assert_abs_diff_eq!(
304            *value,
305            forward
306                .iter()
307                .enumerate()
308                .map(|(i, &j)| cost_matrix[i][j])
309                .sum::<T>(),
310            epsilon = T::epsilon()
311        );
312    }

314    fn compare_with_brute<T: Value + Debug>(cost_matrix: &[Vec<T>], result: &HungarianResult<T>) {
315        let h = cost_matrix.len();
316        let w = cost_matrix[0].len();
317        let value = permutations((0..w).collect_vec())
318            .into_iter()
319            .map(|v| calculate_score(cost_matrix, v[..h].iter().copied()))
320            .min_by(|x, y| x.partial_cmp(y).unwrap())
321            .unwrap();
322        assert_eq!(value, result.value);
323    }
324
325    fn calculate_score<T: Value + Debug>(
326        cost_matrix: &[Vec<T>],
327        perm: impl IntoIterator<Item = usize>,
328    ) -> T {
329        perm.into_iter()
330            .enumerate()
331            .map(|(i, j)| cost_matrix[i][j])
332            .sum::<T>()
333    }
