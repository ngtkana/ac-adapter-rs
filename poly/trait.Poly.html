Expand description

自由関数 poly_add, poly_sub を呼び出します。

Examples

use poly::Poly;

// 成分ごとの和を計算します。
let a = vec![1, 2];
let b = vec![10];
let c = a.into_iter().poly_add(b).collect::<Vec<_>>();
assert_eq!(c, vec![11, 2]);

let a = vec![1, 2];
let b = vec![10];
let c = a.into_iter().poly_sub(b).collect::<Vec<_>>();
assert_eq!(c, vec![-9, 2]);
